  \}
  for (j = 0; j < logN2; j++)
  \{
    if (a[j] != 0)
    y[l] += (1 << ((logN2 -\/ 1) -\/ j));
  \}
  y[l] = y[l] >> 1;
 \} \end{DoxyPre}
