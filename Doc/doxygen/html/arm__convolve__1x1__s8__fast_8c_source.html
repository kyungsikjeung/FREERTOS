54arm_status arm_convolve_1x1_s8_fast(const cmsis_nn_context *ctx,
55 const cmsis_nn_conv_params *conv_params,
56 const cmsis_nn_per_channel_quant_params *quant_params,
57 const cmsis_nn_dims *input_dims,
58 const q7_t *input_data,
59 const cmsis_nn_dims *filter_dims,
60 const q7_t *filter_data,
61 const cmsis_nn_dims *bias_dims,
62 const int32_t *bias_data,
63 const cmsis_nn_dims *output_dims,
64 q7_t *output_data)
65{
66 if (input_dims->c % 4 != 0 || conv_params->padding.w != 0 || conv_params->padding.h != 0 ||
67 conv_params->stride.w != 1 || conv_params->stride.h != 1)
68 {
69 return ARM_MATH_SIZE_MISMATCH;
70 }
71
72 (void)ctx;
73 (void)filter_dims;
74 (void)bias_dims;
75
76#if defined(ARM_MATH_MVEI)
77
78 const int32_t col_len = input_dims->w * input_dims->h * input_dims->n;
79 const int32_t output_ch = output_dims->c;
80 const int32_t input_ch = input_dims->c;
81 const int32_t input_offset = conv_params->input_offset;
82 const int32_t out_offset = conv_params->output_offset;
83 const int32_t out_activation_min = conv_params->activation.min;
84 const int32_t out_activation_max = conv_params->activation.max;
85 int32_t *output_mult = quant_params->multiplier;
86 int32_t *output_shift = quant_params->shift;
87
88 for (int i_items = 0; i_items <= (col_len - 4); i_items += 4)
89 {
90
91 output_data = arm_nn_mat_mul_core_4x_s8(input_ch,
92 input_ch,
93 input_data + i_items * input_ch,
94 filter_data,
95 output_ch,
96 conv_params,
97 quant_params,
98 bias_data,
99 output_data);
100 }
101
102 /* Handle left over elements */
103 for (int i_items = (col_len & ~0x3); i_items < col_len; i_items++)
104 {
105 for (int i_out_ch = 0; i_out_ch < output_ch; i_out_ch++)
106 {
107 int32_t sum_row = 0;
108 int32_t acc;
109 (void)arm_nn_mat_mul_core_1x_s8(
110 input_ch, input_data + i_items * input_ch, filter_data + i_out_ch * input_ch, &sum_row, &acc);
111 if (bias_data)
112 {
113 acc += bias_data[i_out_ch];
114 }
115 sum_row = (sum_row * input_offset);
116 acc += sum_row;
117 acc = arm_nn_requantize(acc, output_mult[i_out_ch], output_shift[i_out_ch]);
118 acc += out_offset;
119
120 acc = MAX(acc, out_activation_min);
121 acc = MIN(acc, out_activation_max);
122 *output_data++ = acc;
123 }
124 }
125
126#else
127 /* Run the following code as reference implementation for Cortex-M processors with or without DSP extension */
128
129 const int32_t lhs_rows = input_dims->w * input_dims->h * input_dims->n;
130 const int32_t rhs_rows = output_dims->c;
131 const int32_t rhs_cols = input_dims->c;
132
133 arm_nn_mat_mult_nt_t_s8(input_data,
134 filter_data,
135 bias_data,
136 output_data,
137 quant_params->multiplier,
138 quant_params->shift,
139 lhs_rows,
140 rhs_rows,
141 rhs_cols,
142 conv_params->input_offset,
143 conv_params->output_offset,
144 conv_params->activation.min,
145 conv_params->activation.max);
146
147#endif
148
149 /* Return to application */
150 return ARM_MATH_SUCCESS;
151}
