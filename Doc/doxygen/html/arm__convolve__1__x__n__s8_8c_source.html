50arm_status arm_convolve_1_x_n_s8(const cmsis_nn_context *ctx,
51 const cmsis_nn_conv_params *conv_params,
52 const cmsis_nn_per_channel_quant_params *quant_params,
53 const cmsis_nn_dims *input_dims,
54 const q7_t *input_data,
55 const cmsis_nn_dims *filter_dims,
56 const q7_t *filter_data,
57 const cmsis_nn_dims *bias_dims,
58 const int32_t *bias_data,
59 const cmsis_nn_dims *output_dims,
60 q7_t *output_data)
61{
62 (void)bias_dims;
63 arm_status status = ARM_MATH_SUCCESS;
64 if (output_dims->w % 4 != 0)
65 {
66 status = ARM_MATH_SIZE_MISMATCH;
67 goto out;
68 }
69
70#if defined(ARM_MATH_MVEI)
71 (void)ctx;
72
73 const uint16_t input_x = input_dims->w;
74 const uint16_t kernel_x = filter_dims->w;
75 const uint16_t output_x = output_dims->w;
76 const uint16_t output_ch = output_dims->c;
77 const uint16_t input_ch = input_dims->c;
78 const uint16_t pad_x = conv_params->padding.w;
79 const uint16_t stride_x = conv_params->stride.w;
80
81 const int32_t input_offset = conv_params->input_offset;
82 const int32_t out_offset = conv_params->output_offset;
83 const int32_t out_activation_min = conv_params->activation.min;
84 const int32_t out_activation_max = conv_params->activation.max;
85 int32_t *output_mult = quant_params->multiplier;
86 int32_t *output_shift = quant_params->shift;
87
88 for (int i_out_x = 0; i_out_x <= (output_x - 4); i_out_x += 4)
89 {
90 int32_t input_begin_idx[4];
91 int32_t ker_begin_idx[4];
92 int32_t ker_end_idx[4];
93
94 for (int i = 0; i < 4; i++)
95 {
96 const int32_t est_input_x_idx = stride_x * (i_out_x + i) - pad_x;
97 input_begin_idx[i] = MAX(0, est_input_x_idx);
98 ker_begin_idx[i] = MAX(0, -est_input_x_idx);
99 ker_end_idx[i] = MIN(kernel_x, input_x - est_input_x_idx);
100 }
101
102 if ((ker_begin_idx[0] != 0) || (ker_end_idx[3] != kernel_x))
103 {
104 for (int i_out_ch = 0; i_out_ch < output_ch; i_out_ch++)
105 {
106 int32x4_t s_offset;
107 int32_t acc[4];
108 {
109 int32_t sum_row[4];
110
111 (void)arm_nn_mat_mul_core_1x_s8((ker_end_idx[0] - ker_begin_idx[0]) * input_ch,
112 input_data + input_begin_idx[0] * input_ch,
113 filter_data + (input_ch * kernel_x * i_out_ch) +
114 (ker_begin_idx[0] * input_ch),
115 &sum_row[0],
116 &acc[0]);
117 (void)arm_nn_mat_mul_core_1x_s8((ker_end_idx[1] - ker_begin_idx[1]) * input_ch,
118 input_data + input_begin_idx[1] * input_ch,
119 filter_data + (input_ch * kernel_x * i_out_ch) +
120 (ker_begin_idx[1] * input_ch),
121 &sum_row[1],
122 &acc[1]);
123
124 (void)arm_nn_mat_mul_core_1x_s8((ker_end_idx[2] - ker_begin_idx[2]) * input_ch,
125 input_data + input_begin_idx[2] * input_ch,
126 filter_data + (input_ch * kernel_x * i_out_ch) +
127 (ker_begin_idx[2] * input_ch),
128 &sum_row[2],
129 &acc[2]);
130
131 (void)arm_nn_mat_mul_core_1x_s8((ker_end_idx[3] - ker_begin_idx[3]) * input_ch,
132 input_data + input_begin_idx[3] * input_ch,
133 filter_data + (input_ch * kernel_x * i_out_ch) +
134 (ker_begin_idx[3] * input_ch),
135 &sum_row[3],
136 &acc[3]);
137
138 s_offset = vldrwq_s32(sum_row);
139 }
140 int32x4_t res = vldrwq_s32(acc);
141 s_offset = vmulq_n_s32(s_offset, input_offset);
142 res = vaddq_s32(res, s_offset);
143 if (bias_data)
144 {
145 res = vaddq_n_s32(res, bias_data[i_out_ch]);
146 }
147 res = arm_requantize_mve(res, output_mult[i_out_ch], output_shift[i_out_ch]);
148 res = vaddq_n_s32(res, out_offset);
149
150 res = vmaxq_s32(res, vdupq_n_s32(out_activation_min));
151 res = vminq_s32(res, vdupq_n_s32(out_activation_max));
152
153 const uint32x4_t scatter_offset = {0, output_ch, output_ch * 2, output_ch * 3};
154 vstrbq_scatter_offset_s32(output_data, scatter_offset, res);
155 output_data++;
156 }
157 output_data += (3 * output_ch);
158 }
159 else
160 {
161 output_data = arm_nn_mat_mul_core_4x_s8(kernel_x * input_ch,
162 stride_x * input_ch,
163 input_data + input_begin_idx[0] * input_ch,
164 filter_data,
165 output_ch,
166 conv_params,
167 quant_params,
168 bias_data,
169 output_data);
170 }
171 }
172
173#else
174 status = arm_convolve_s8(ctx,
175 conv_params,
176 quant_params,
177 input_dims,
178 input_data,
179 filter_dims,
180 filter_data,
181 bias_dims,
182 bias_data,
183 output_dims,
184 output_data);
185#endif
186
187out:
188 /* Return to application */
189 return status;
190}

201}
