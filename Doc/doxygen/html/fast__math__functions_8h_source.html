256__STATIC_FORCEINLINE arm_status arm_sqrt_f32(
257 const float32_t in,
258 float32_t * pOut)
259 {
260 if (in >= 0.0f)
261 {
262#if defined ( __CC_ARM )
263 #if defined __TARGET_FPU_VFP
264 *pOut = __sqrtf(in);
265 #else
266 *pOut = sqrtf(in);
267 #endif
268
269#elif defined ( __ICCARM__ )
270 #if defined __ARMVFP__
271 __ASM("VSQRT.F32 %0,%1" : "=t"(*pOut) : "t"(in));
272 #else
273 *pOut = sqrtf(in);
274 #endif
275
276#else
277 *pOut = sqrtf(in);
278#endif
279
280 return (ARM_MATH_SUCCESS);
281 }
282 else
283 {
284 *pOut = 0.0f;
285 return (ARM_MATH_ARGUMENT_ERROR);
286 }
287 }
