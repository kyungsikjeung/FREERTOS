43/*
44 * Basic s16 convolution function.
45 *
46 * Refer header file for details. Optimal use case for the DSP/MVE implementation is when input and output channels
47 * are multiples of 4 or atleast greater than 4.
48 *
49 */
50
51arm_status arm_convolve_fast_s16(const cmsis_nn_context *ctx,
52 const cmsis_nn_conv_params *conv_params,
53 const cmsis_nn_per_channel_quant_params *quant_params,
54 const cmsis_nn_dims *input_dims,
55 const q15_t *input_data,
56 const cmsis_nn_dims *filter_dims,
57 const q7_t *filter_data,
58 const cmsis_nn_dims *bias_dims,
59 const int64_t *bias_data,
60 const cmsis_nn_dims *output_dims,
61 q15_t *output_data)
62{
63 (void)bias_dims;
64 if (filter_dims->w * filter_dims->h * input_dims->c >= 512)
65 {
66 return ARM_MATH_SIZE_MISMATCH;
67 }
68
69 if (ctx->buf == NULL && arm_convolve_s8_get_buffer_size(input_dims, filter_dims) > 0)
70 {
71 return ARM_MATH_ARGUMENT_ERROR;
72 }
73 q15_t *buffer_a = (q15_t *)ctx->buf;
74
75 const int32_t input_batches = input_dims->n;
76 const int32_t input_x = input_dims->w;
77 const int32_t input_y = input_dims->h;
78 const int32_t input_ch = input_dims->c;
79 const int32_t kernel_x = filter_dims->w;
80 const int32_t kernel_y = filter_dims->h;
81 const int32_t output_x = output_dims->w;
82 const int32_t output_y = output_dims->h;
83 const int32_t output_ch = output_dims->c;
84
85 const int32_t pad_x = conv_params->padding.w;
86 const int32_t pad_y = conv_params->padding.h;
87 const int32_t stride_x = conv_params->stride.w;
88 const int32_t stride_y = conv_params->stride.h;
89
90 const int16_t out_activation_min = conv_params->activation.min;
91 const int16_t out_activation_max = conv_params->activation.max;
92 int32_t *output_mult = quant_params->multiplier;
93 int32_t *output_shift = quant_params->shift;
94
95 for (int i_batch = 0; i_batch < input_batches; i_batch++)
96 {
97#if defined(ARM_MATH_DSP) && !defined(ARM_MATH_MVEI)
98 /* Generate two columns from the input tensor a GEMM computation */
99 q15_t *two_column_buf = buffer_a;
100 q15_t *out = output_data;
101 /* This part implements the im2col function */
102 for (int32_t i_out_y = 0; i_out_y < output_y; i_out_y++)
103 {
104 for (int32_t i_out_x = 0; i_out_x < output_x; i_out_x++)
105 {
106 for (int32_t i_ker_y = i_out_y * stride_y - pad_y; i_ker_y < i_out_y * stride_y - pad_y + kernel_y;
107 i_ker_y++)
108 {
109 for (int32_t i_ker_x = i_out_x * stride_x - pad_x; i_ker_x < i_out_x * stride_x - pad_x + kernel_x;
110 i_ker_x++)
111 {
112 if (i_ker_y < 0 || i_ker_y >= input_y || i_ker_x < 0 || i_ker_x >= input_x)
113 {
114 /* Filling 0 for out-of-bound paddings */
115 arm_memset_q7((q7_t *)two_column_buf, 0, sizeof(q15_t) * input_ch);
116 }
117 else
118 {
119 arm_memcpy_q7((q7_t *)two_column_buf,
120 (const q7_t *)(input_data + (i_ker_y * input_x + i_ker_x) * input_ch),
121 input_ch * sizeof(q15_t));
122 }
123 two_column_buf += input_ch;
124 }
125 }
126 /* Computation is filed for every 2 columns */
127 if (two_column_buf == buffer_a + 2 * input_ch * kernel_y * kernel_x)
128 {
129 out = arm_nn_mat_mult_kernel_s16(filter_data,
130 buffer_a,
131 output_ch,
132 output_shift,
133 output_mult,
134 out_activation_min,
135 out_activation_max,
136 (input_ch * kernel_y * kernel_x),
137 bias_data,
138 out);
139
140 /* Counter reset */
141 two_column_buf = buffer_a;
142 }
143 }
144 }
145
146 /* Left-over because odd number of output pixels */
147 if (two_column_buf != buffer_a)
148 {
149 const q7_t *ker_a = filter_data;
150 int i;
151
152 for (i = 0; i < output_ch; i++)
153 {
154 /* Init the accumulator*/
155 q31_t sum = 0;
156
157 /* Point to the beginning of the im2col buffer where the input is available as a rearranged column */
158 const q15_t *ip_as_col = buffer_a;
159
160 /* 4 multiply and accumulates are done in one loop. */
161 uint16_t col_count = (input_ch * kernel_y * kernel_x) >> 2;
162
163 while (col_count)
164 {
165 q31_t ker_a1, ker_a2;
166 q31_t ip_b1, ip_b2;
167
168 ker_a = read_and_pad(ker_a, &ker_a1, &ker_a2);
169
170 ip_b1 = arm_nn_read_q15x2_ia(&ip_as_col);
171 sum = __SMLAD(ker_a1, ip_b1, sum);
172 ip_b2 = arm_nn_read_q15x2_ia(&ip_as_col);
173 sum = __SMLAD(ker_a2, ip_b2, sum);
174
175 col_count--;
176 }
177 /* Handle left over mac */
178 col_count = input_ch * kernel_y * kernel_x & 0x3;
179 while (col_count)
180 {
181 q7_t ker_a1 = *ker_a++;
182 q15_t ip_b1 = *ip_as_col++;
183 sum += ker_a1 * ip_b1;
184 col_count--;
185 }
186 if (bias_data)
187 {
188 q31_t reduced_multiplier = REDUCE_MULTIPLIER(output_mult[i]);
189 q63_t acc_64 = sum + bias_data[i];
190 sum = arm_nn_requantize_s64(acc_64, reduced_multiplier, output_shift[i]);
191 }
192 else
193 {
194 sum = arm_nn_requantize(sum, output_mult[i], output_shift[i]);
195 }
196 sum = MAX(sum, out_activation_min);
197 sum = MIN(sum, out_activation_max);
198 *out++ = (q15_t)sum;
199 }
200 }
201#else
202 (void)input_data;
203 (void)output_data;
204 (void)bias_data;
205 (void)filter_data;
206 (void)buffer_a;
207 (void)kernel_x;
208 (void)kernel_y;
209 (void)pad_x;
210 (void)pad_y;
211 (void)stride_x;
212 (void)stride_y;
213 (void)out_activation_min;
214 (void)out_activation_max;
215 (void)output_mult;
216 (void)output_shift;
217 return ARM_MATH_ARGUMENT_ERROR;
218#endif
219 /* Advance to the next batch */
220 input_data += (input_x * input_y * input_ch);
221 output_data += (output_x * output_y * output_ch);
222 }
223
224 /* Return to application */
225 return ARM_MATH_SUCCESS;
226}

228int32_t arm_convolve_fast_s16_get_buffer_size(const cmsis_nn_dims *input_dims, const cmsis_nn_dims *filter_dims)
229{
230#if defined(ARM_MATH_DSP) && !defined(ARM_MATH_MVEI)
231 return (2 * input_dims->c * filter_dims->w * filter_dims->h) * (int32_t)sizeof(int16_t);
232#else
233 (void)input_dims;
234 (void)filter_dims;
235 return 0;
236#endif
237}
