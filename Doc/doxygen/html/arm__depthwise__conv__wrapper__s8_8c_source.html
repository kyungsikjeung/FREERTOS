49arm_status arm_depthwise_conv_wrapper_s8(const cmsis_nn_context *ctx,
50 const cmsis_nn_dw_conv_params *dw_conv_params,
51 const cmsis_nn_per_channel_quant_params *quant_params,
52 const cmsis_nn_dims *input_dims,
53 const q7_t *input,
54 const cmsis_nn_dims *filter_dims,
55 const q7_t *filter,
56 const cmsis_nn_dims *bias_dims,
57 const int32_t *bias,
58 const cmsis_nn_dims *output_dims,
59 q7_t *output)
60{
61 arm_status status = ARM_MATH_SUCCESS;
62 if (1 == dw_conv_params->ch_mult && input_dims->n == 1 && dw_conv_params->dilation.w == 1 &&
63 dw_conv_params->dilation.h == 1)
64 {
65#if !defined(ARM_MATH_MVEI)
66 if ((filter_dims->w == 3) && (filter_dims->h == 3) && (dw_conv_params->padding.h <= 1) &&
67 (dw_conv_params->padding.w <= 1))
68 {
69 status = arm_depthwise_conv_3x3_s8(ctx,
70 dw_conv_params,
71 quant_params,
72 input_dims,
73 input,
74 filter_dims,
75 filter,
76 bias_dims,
77 bias,
78 output_dims,
79 output);
80 }
81 else
82#endif
83 {
84 status = arm_depthwise_conv_s8_opt(ctx,
85 dw_conv_params,
86 quant_params,
87 input_dims,
88 input,
89 filter_dims,
90 filter,
91 bias_dims,
92 bias,
93 output_dims,
94 output);
95 }
96 }
97 else
98 {
99 status = arm_depthwise_conv_s8(ctx,
100 dw_conv_params,
101 quant_params,
102 input_dims,
103 input,
104 filter_dims,
105 filter,
106 bias_dims,
107 bias,
108 output_dims,
109 output);
110 }
111
112 /* Return to application */
113 return status;
114}

116int32_t arm_depthwise_conv_wrapper_s8_get_buffer_size(const cmsis_nn_dw_conv_params *dw_conv_params,
117 const cmsis_nn_dims *input_dims,
118 const cmsis_nn_dims *filter_dims,
119 const cmsis_nn_dims *output_dims)
120{
121 (void)dw_conv_params;
122 int32_t size = 0;
123
124 if (input_dims->c == output_dims->c && input_dims->n == 1 && dw_conv_params->dilation.w == 1 &&
125 dw_conv_params->dilation.h == 1)
126 {
127 size = arm_depthwise_conv_s8_opt_get_buffer_size(input_dims, filter_dims);
128 }
129
130 return size;
131}
