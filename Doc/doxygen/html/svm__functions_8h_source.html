65__STATIC_INLINE float32_t arm_exponent_f32(float32_t x, int32_t nb)
66{
67 float32_t r = x;
68 nb --;
69 while(nb > 0)
70 {
71 r = r * x;
72 nb--;
73 }
74 return(r);
75}
