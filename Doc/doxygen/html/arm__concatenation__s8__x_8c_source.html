49void arm_concatenation_s8_x(const int8_t *input,
50 const uint16_t input_x,
51 const uint16_t input_y,
52 const uint16_t input_z,
53 const uint16_t input_w,
54 int8_t *output,
55 const uint16_t output_x,
56 const uint32_t offset_x)
57{
58 const uint32_t num_iterations = input_y * input_z * input_w;
59
60 output += offset_x;
61
62 uint32_t i;
63
64 // Copy per row
65 for (i = 0; i < num_iterations; ++i)
66 {
67 arm_memcpy_q7(output, input, input_x);
68 input += input_x;
69 output += output_x;
70 }
71}
