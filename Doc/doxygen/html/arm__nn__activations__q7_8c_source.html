59void arm_nn_activations_direct_q7(q7_t *data, uint16_t size, uint16_t int_width, arm_nn_activation_type type)
60{
61 uint16_t i = size;
62 q7_t *pIn = data;
63 q7_t *pOut = data;
64 q7_t in;
65 q7_t out;
66 uint16_t shift_size = 3 - int_width;
67 const q7_t *lookup_table;
68 switch (type)
69 {
70 case ARM_SIGMOID:
71 lookup_table = sigmoidTable_q7;
72 break;
73 case ARM_TANH:
74 default:
75 lookup_table = tanhTable_q7;
76 break;
77 }
78 while (i)
79 {
80 in = *pIn++;
81 out = lookup_table[(uint8_t)(in >> shift_size)];
82 *pOut++ = out;
83 i--;
84 }
85}
