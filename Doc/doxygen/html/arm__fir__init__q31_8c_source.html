74void arm_fir_init_q31(
75 arm_fir_instance_q31 * S,
76 uint16_t numTaps,
77 const q31_t * pCoeffs,
78 q31_t * pState,
79 uint32_t blockSize)
80{
81 /* Assign filter taps */
82 S->numTaps = numTaps;
83
84 /* Assign coefficient pointer */
85 S->pCoeffs = pCoeffs;
86
87 /* Clear state buffer. The size is always (blockSize + numTaps - 1) */
88 #if defined(ARM_MATH_MVEI) && !defined(ARM_MATH_AUTOVECTORIZE)
89 memset(pState, 0, (numTaps + (blockSize - 1U) + 2*ROUND_UP(blockSize, 4)) * sizeof(q31_t));
90 #else
91 memset(pState, 0, (numTaps + (blockSize - 1U)) * sizeof(q31_t));
92 #endif
93
94 /* Assign state pointer */
95 S->pState = pState;
96}
