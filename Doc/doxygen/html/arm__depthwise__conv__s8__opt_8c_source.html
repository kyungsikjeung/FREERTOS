44/*
45 * Optimized s8 depthwise convolution function with constraint that in_channel equals out_channel
46 *
47 * Refer prototype header file for details.
48 *
49 */
50
51arm_status arm_depthwise_conv_s8_opt(const cmsis_nn_context *ctx,
52 const cmsis_nn_dw_conv_params *dw_conv_params,
53 const cmsis_nn_per_channel_quant_params *quant_params,
54 const cmsis_nn_dims *input_dims,
55 const q7_t *input,
56 const cmsis_nn_dims *filter_dims,
57 const q7_t *kernel,
58 const cmsis_nn_dims *bias_dims,
59 const int32_t *bias,
60 const cmsis_nn_dims *output_dims,
61 q7_t *output)
62{
63
64 const int32_t input_ch = input_dims->c;
65 const int32_t output_ch = output_dims->c;
66
67 /* Check input constraints input_ch == output_ch */
68 if (input_ch != output_ch)
69 {
70 return ARM_MATH_SIZE_MISMATCH;
71 }
72
73 if (ctx->buf == NULL && arm_depthwise_conv_s8_opt_get_buffer_size(input_dims, filter_dims) > 0)
74 {
75 return ARM_MATH_ARGUMENT_ERROR;
76 }
77#ifdef ARM_MATH_DSP
78 const int32_t input_x = input_dims->w;
79 const int32_t input_y = input_dims->h;
80 const int32_t kernel_x = filter_dims->w;
81 const int32_t kernel_y = filter_dims->h;
82 const int32_t pad_x = dw_conv_params->padding.w;
83 const int32_t pad_y = dw_conv_params->padding.h;
84 const int32_t stride_x = dw_conv_params->stride.w;
85 const int32_t stride_y = dw_conv_params->stride.h;
86 const int32_t *output_shift = quant_params->shift;
87 const int32_t *output_mult = quant_params->multiplier;
88 const int32_t output_x = output_dims->w;
89 const int32_t output_y = output_dims->h;
90 const int32_t output_offset = dw_conv_params->output_offset;
91 const int32_t input_offset = dw_conv_params->input_offset;
92 const int32_t output_activation_min = dw_conv_params->activation.min;
93 const int32_t output_activation_max = dw_conv_params->activation.max;
94 q15_t *buffer_a = (q15_t *)ctx->buf;
95
96#ifdef ARM_MATH_MVEI
97 (void)bias_dims;
98 /* Generate two columns from the input tensor */
99 q7_t *lhs_buffer = (q7_t *)buffer_a;
100 q7_t *out = output;
101 int padded = 0;
102 int buffer_count = 0;
103 const int32_t kernel_size = kernel_x * kernel_y;
104
105 /* This part implements the im2col function */
106 for (int i_out_y = 0, base_idx_y = -pad_y; i_out_y < output_y; base_idx_y += stride_y, i_out_y++)
107 {
108 for (int i_out_x = 0, base_idx_x = -pad_x; i_out_x < output_x; base_idx_x += stride_x, i_out_x++)
109 {
110 for (int i_ker_y = base_idx_y; i_ker_y < base_idx_y + kernel_y; i_ker_y++)
111 {
112 for (int i_ker_x = base_idx_x; i_ker_x < base_idx_x + kernel_x; i_ker_x++)
113 {
114 if (i_ker_y < 0 || i_ker_y >= input_y || i_ker_x < 0 || i_ker_x >= input_x)
115 {
116 arm_memset_q7(lhs_buffer, (int8_t)-input_offset, (uint32_t)input_ch);
117 padded = 1;
118 }
119 else
120 {
121 arm_memcpy_q7(lhs_buffer, input + (i_ker_y * input_x + i_ker_x) * input_ch, (uint32_t)input_ch);
122 }
123 lhs_buffer += input_ch;
124 }
125 }
126 buffer_count++;
127
128 if (buffer_count == 4)
129 {
130 lhs_buffer = (q7_t *)buffer_a;
131 if (padded == 0)
132 {
133 out = arm_nn_depthwise_conv_nt_t_s8(lhs_buffer,
134 kernel,
135 input_offset,
136 input_ch,
137 output_shift,
138 output_mult,
139 output_offset,
140 output_activation_min,
141 output_activation_max,
142 kernel_size,
143 bias,
144 out);
145 }
146 else
147 {
148 out = arm_nn_depthwise_conv_nt_t_padded_s8(lhs_buffer,
149 kernel,
150 input_offset,
151 input_ch,
152 output_shift,
153 output_mult,
154 output_offset,
155 output_activation_min,
156 output_activation_max,
157 kernel_size,
158 bias,
159 out);
160 padded = 0;
161 }
162 buffer_count = 0;
163 }
164 }
165 }
166
167 /* Handle left over buffers */
168 lhs_buffer = (q7_t *)buffer_a;
169
170 for (int i_buf = 0; i_buf < buffer_count; i_buf++)
171 {
172 int32_t loop_count = (input_ch + 3) / 4;
173
174 int32_t num_ch_to_process = input_ch;
175 for (int i_loop_cnt = 0, offset = 0; i_loop_cnt < loop_count; num_ch_to_process -= 4, offset += 4, i_loop_cnt++)
176 {
177 const int8_t *col_0 = lhs_buffer + (kernel_size * input_ch * i_buf) + offset;
178 const int8_t *row_0 = kernel + offset;
179 int32x4_t out_0 = vldrwq_s32(&bias[offset]);
180
181 for (int i_ker = 0; i_ker < kernel_size; i_ker++)
182 {
183 const int32x4_t ker_0 = vldrbq_s32(row_0);
184
185 int32x4_t ip_0 = vldrbq_s32(col_0);
186 ip_0 = vaddq_n_s32(ip_0, input_offset);
187 out_0 += vmulq_s32(ip_0, ker_0);
188
189 col_0 += input_ch;
190 row_0 += input_ch;
191 }
192
193 const int32x4_t mult = vldrwq_s32(&output_mult[offset]);
194 const int32x4_t shift = vldrwq_s32(&output_shift[offset]);
195
196 out_0 = arm_requantize_mve_32x4(out_0, mult, shift);
197 out_0 = vaddq_n_s32(out_0, output_offset);
198 out_0 = vmaxq_s32(out_0, vdupq_n_s32(output_activation_min));
199 out_0 = vminq_s32(out_0, vdupq_n_s32(output_activation_max));
200 mve_pred16_t p = vctp32q((uint32_t)num_ch_to_process);
201 vstrbq_p_s32(out, out_0, p);
202
203 out += 4;
204 }
205
206 const int tail_ch = input_ch & 0x3;
207 if (tail_ch != 0)
208 {
209 out -= (4 - tail_ch);
210 }
211 }
212
213#else // ARM_MATH_DSP
214 (void)bias_dims;
215 /* Run the following code in cores using DSP extension */
216 q15_t *const col_buffer_start = buffer_a;
217 q15_t *col_buffer = col_buffer_start;
218 const int32_t *const bias_start_pos = bias;
219 const q31_t *const out_mult_start_pos = output_mult;
220 const q31_t *const out_shift_start_pos = output_shift;
221 uint16_t row_count;
222 uint16_t row_shift;
223
224 for (int i_out_y = 0; i_out_y < output_y; i_out_y++)
225 {
226 const int16_t base_idx_y = (i_out_y * stride_y) - pad_y;
227 for (int i_out_x = 0; i_out_x < output_x; i_out_x++)
228 {
229 const int16_t base_idx_x = (i_out_x * stride_x) - pad_x;
230
231 /* Out of bounds is only considered for the y axis as it provides a contiguous zero'ing opportunity than
232 along the x axis */
233 const int ker_y_start = MAX(0, -base_idx_y);
234 /* Condition for kernel end dimension: (base_idx_y + ker_y_end) < input_y */
235 const int ker_y_end = MIN(kernel_y, input_y - base_idx_y);
236
237 int32_t index = 0;
238 if (ker_y_start != 0)
239 {
240 memset(&col_buffer[index], 0, (kernel_x * input_ch) * ker_y_start * sizeof(q15_t));
241 index += (kernel_x * input_ch) * ker_y_start;
242 }
243
244 for (int i_ker_y = ker_y_start; i_ker_y < ker_y_end; i_ker_y++)
245 {
246 const int32_t idx_y = base_idx_y + i_ker_y;
247
248 for (int i_ker_x = 0; i_ker_x < kernel_x; i_ker_x++)
249 {
250 const int32_t idx_x = base_idx_x + i_ker_x;
251 if (idx_x < 0 || idx_x >= input_x)
252 {
253 memset(&col_buffer[index], 0, input_ch * sizeof(q15_t));
254 }
255 else
256 {
257 arm_q7_to_q15_with_offset((q7_t *)input + (idx_y * input_x + idx_x) * input_ch,
258 &col_buffer[index],
259 input_ch,
260 input_offset);
261 }
262 index += input_ch;
263 }
264 }
265
266 const int diff = kernel_y - ker_y_end;
267 if (diff != 0)
268 {
269 memset(&col_buffer[index], 0, (kernel_x * input_ch) * diff * sizeof(q15_t));
270 }
271
272 row_count = output_ch / 4;
273 row_shift = 0;
274 bias = bias_start_pos;
275 output_mult = out_mult_start_pos;
276 output_shift = out_shift_start_pos;
277
278 while (row_count)
279 {
280 q31_t sum = *bias++;
281 q31_t sum_2 = *bias++;
282 q31_t sum_3 = *bias++;
283 q31_t sum_4 = *bias++;
284
285 uint16_t col_count = (kernel_x * kernel_y) / 2;
286 q15_t *col_pos = col_buffer_start + row_shift;
287 const q7_t *row_pos = kernel + row_shift;
288 row_shift += 4;
289
290 while (col_count)
291 {
292 /* General idea is to read 4 + 4 (input, kernel) pair and re-arrange them in the right order to
293 use in a SMLAD instruction . One run of this loop produces 4 partial outputs with 8 MACs. */
294 /* Note: variable names can be improved here to align with rows and columns. */
295 q31_t ip_a1, ip_a2, ip_b1, ip_b2, op_a, op_b, op_c;
296 /* Read 4 weights */
297 ip_b1 = arm_nn_read_q7x4(row_pos);
298 ip_a1 = arm_nn_read_q7x4(row_pos + input_ch);
299 op_a = arm_nn_read_q15x2(col_pos);
300 op_b = arm_nn_read_q15x2(col_pos + input_ch);
301
302 ip_a2 = __SXTB16(ip_b1);
303 ip_b1 = __SXTB16(__ROR(ip_b1, 8));
304
305 ip_b2 = __SXTB16(ip_a1);
306 ip_a1 = __SXTB16(__ROR(ip_a1, 8));
307
308 op_c = __PKHBT(op_b, op_a, 16);
309 op_a = __PKHTB(op_b, op_a, 16);
310 op_b = __PKHBT(ip_b2, ip_a2, 16);
311 sum = __SMLAD(op_c, op_b, sum);
312
313 op_b = __PKHBT(ip_b1, ip_a1, 16);
314 sum_2 = __SMLAD(op_a, op_b, sum_2);
315
316 op_a = arm_nn_read_q15x2(col_pos + 2);
317 op_b = arm_nn_read_q15x2(col_pos + input_ch + 2);
318
319 op_c = __PKHBT(op_b, op_a, 16);
320 op_a = __PKHTB(op_b, op_a, 16);
321 op_b = __PKHTB(ip_a2, ip_b2, 16);
322 sum_3 = __SMLAD(op_c, op_b, sum_3);
323
324 op_b = __PKHTB(ip_a1, ip_b1, 16);
325 sum_4 = __SMLAD(op_a, op_b, sum_4);
326
327 row_pos += input_ch << 1;
328 col_pos += input_ch << 1;
329 col_count--;
330 }
331
332 col_count = (kernel_x * kernel_y) & 0x1;
333 while (col_count)
334 {
335 sum += row_pos[0] * col_pos[0];
336 sum_2 += row_pos[1] * col_pos[1];
337 sum_3 += row_pos[2] * col_pos[2];
338 sum_4 += row_pos[3] * col_pos[3];
339
340 row_pos += input_ch;
341 col_pos += input_ch;
342
343 col_count--;
344 }
345 sum = arm_nn_requantize(sum, *output_mult++, *output_shift++);
346 sum += output_offset;
347 sum = MAX(sum, output_activation_min);
348 sum = MIN(sum, output_activation_max);
349 *output++ = (q7_t)sum;
350
351 sum_2 = arm_nn_requantize(sum_2, *output_mult++, *output_shift++);
352 sum_2 += output_offset;
353 sum_2 = MAX(sum_2, output_activation_min);
354 sum_2 = MIN(sum_2, output_activation_max);
355 *output++ = (q7_t)sum_2;
356 sum_3 = arm_nn_requantize(sum_3, *output_mult++, *output_shift++);
357 sum_3 += output_offset;
358 sum_3 = MAX(sum_3, output_activation_min);
359 sum_3 = MIN(sum_3, output_activation_max);
360 *output++ = (q7_t)sum_3;
361
362 sum_4 = arm_nn_requantize(sum_4, *output_mult++, *output_shift++);
363 sum_4 += output_offset;
364 sum_4 = MAX(sum_4, output_activation_min);
365 sum_4 = MIN(sum_4, output_activation_max);
366 *output++ = (q7_t)sum_4;
367
368 row_count--;
369 }
370
371 row_count = output_ch & 0x3;
372 while (row_count)
373 {
374 q15_t *col_pos = col_buffer_start + row_shift;
375 const q7_t *row_pos = kernel + row_shift;
376 q31_t sum = *bias++;
377 const uint16_t col_count = (kernel_x * kernel_y);
378 row_shift += 1;
379
380 for (int i = 0; i < col_count; i++)
381 {
382 sum += row_pos[i * input_ch] * col_pos[i * input_ch];
383 }
384 sum = arm_nn_requantize(sum, *output_mult++, *output_shift++);
385 sum += output_offset;
386 sum = MAX(sum, output_activation_min);
387 sum = MIN(sum, output_activation_max);
388 *output++ = (q7_t)sum;
389
390 row_count--;
391 }
392
393 // clear counter and pointers
394 col_buffer = col_buffer_start;
395 }
396 }
397#endif
398#else
399 /* Run the following code as reference implementation for Cortex-M0 and Cortex-M3 */
400 return arm_depthwise_conv_s8(ctx,
401 dw_conv_params,
402 quant_params,
403 input_dims,
404 input,
405 filter_dims,
406 kernel,
407 bias_dims,
408 bias,
409 output_dims,
410 output);
411#endif /* ARM_MATH_MVEI | ARM_MATH_DSP */
412
413 /* Return to application */
414 return ARM_MATH_SUCCESS;
415}

417int32_t arm_depthwise_conv_s8_opt_get_buffer_size(const cmsis_nn_dims *input_dims, const cmsis_nn_dims *filter_dims)
418{
419#if defined(ARM_MATH_MVEI)
420 /* The + 4 accounts for out of bounds read of the lhs buffers in the *_nt_t_* functions. */
421 return (2 * input_dims->c * filter_dims->w * filter_dims->h) * (int32_t)sizeof(int16_t) + 4;
422#elif defined(ARM_MATH_DSP)
423 return (input_dims->c * filter_dims->w * filter_dims->h) * sizeof(int16_t);
424#else
425 (void)input_dims;
426 (void)filter_dims;
427 return 0;
428#endif
