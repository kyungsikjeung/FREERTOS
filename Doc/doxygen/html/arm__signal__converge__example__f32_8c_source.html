184/* ----------------------------------------------------------------------
185* Signal converge test
186* ------------------------------------------------------------------- */
187
188int32_t main(void)
189{
190 uint32_t i;
191 arm_status status;
192 uint32_t index;
193 float32_t minValue;
194
195 /* Initialize the LMSNorm data structure */
196 arm_lms_norm_init_f32(&lmsNorm_instance, NUMTAPS, lmsNormCoeff_f32, lmsStateF32, MU, BLOCKSIZE);
197
198 /* Initialize the FIR data structure */
199 arm_fir_init_f32(&LPF_instance, NUMTAPS, (float32_t *)FIRCoeff_f32, firStateF32, BLOCKSIZE);
200
201 /* ----------------------------------------------------------------------
202 * Loop over the frames of data and execute each of the processing
203 * functions in the system.
204 * ------------------------------------------------------------------- */
205
206 for(i=0; i < NUMFRAMES; i++)
207 {
208 /* Read the input data - uniformly distributed random noise - into wire1 */
209 arm_copy_f32(testInput_f32 + (i * BLOCKSIZE), wire1, BLOCKSIZE);
210
211 /* Execute the FIR processing function. Input wire1 and output wire2 */
212 arm_fir_f32(&LPF_instance, wire1, wire2, BLOCKSIZE);
213
214 /* Execute the LMS Norm processing function*/
215
216 arm_lms_norm_f32(&lmsNorm_instance, /* LMSNorm instance */
217 wire1, /* Input signal */
218 wire2, /* Reference Signal */
219 wire3, /* Converged Signal */
220 err_signal, /* Error Signal, this will become small as the signal converges */
221 BLOCKSIZE); /* BlockSize */
222
223 /* apply overall gain */
224 arm_scale_f32(wire3, 5, wire3, BLOCKSIZE); /* in-place buffer */
225 }
226
227 status = ARM_MATH_SUCCESS;
228
229 /* -------------------------------------------------------------------------------
230 * Test whether the error signal has reached towards 0.
231 * ----------------------------------------------------------------------------- */
232
233 arm_abs_f32(err_signal, err_signal, BLOCKSIZE);
234 arm_min_f32(err_signal, BLOCKSIZE, &minValue, &index);
235
236 if (minValue > DELTA_ERROR)
237 {
238 status = ARM_MATH_TEST_FAILURE;
239 }
240
241 /* ----------------------------------------------------------------------
242 * Test whether the filter coefficients have converged.
243 * ------------------------------------------------------------------- */
244
245 arm_sub_f32((float32_t *)FIRCoeff_f32, lmsNormCoeff_f32, lmsNormCoeff_f32, NUMTAPS);
246
247 arm_abs_f32(lmsNormCoeff_f32, lmsNormCoeff_f32, NUMTAPS);
248 arm_min_f32(lmsNormCoeff_f32, NUMTAPS, &minValue, &index);
249
250 status = (minValue > DELTA_COEFF) ? ARM_MATH_TEST_FAILURE : ARM_MATH_SUCCESS;
251
252 if (status != ARM_MATH_SUCCESS)
253 {
254#if defined (SEMIHOSTING)
255 printf("FAILURE\n");
256#else
257 while (1); /* main function does not return */
258#endif
259 }
260 else
261 {
262#if defined (SEMIHOSTING)
263 printf("SUCCESS\n");
264#else
265 while (1); /* main function does not return */
266#endif
267 }
268
269}
