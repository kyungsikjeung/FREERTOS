50void arm_nn_activations_direct_q15(q15_t *data, uint16_t size, uint16_t int_width, arm_nn_activation_type type)
51{
52 uint16_t i = size;
53 q15_t *pIn = data;
54 q15_t *pOut = data;
55 uint16_t shift_size = 8 + 3 - int_width;
56 uint32_t bit_mask = 0x7FF >> int_width;
57 uint32_t full_frac = bit_mask + 1;
58 const q15_t *lookup_table;
59
60 switch (type)
61 {
62 case ARM_SIGMOID:
63 lookup_table = sigmoidTable_q15;
64 break;
65 case ARM_TANH:
66 default:
67 lookup_table = tanhTable_q15;
68 break;
69 }
70
71 while (i)
72 {
73 q15_t out;
74 q15_t in = *pIn++;
75 q15_t frac = (uint32_t)in & bit_mask;
76 q15_t value = lookup_table[(uint8_t)(in >> shift_size)];
77 if ((in >> shift_size) != 0x7f)
78 {
79 q15_t value2 = lookup_table[(uint8_t)(1 + ((uint8_t)(in >> shift_size)))];
80 /* doing the interpolation here for better accuracy */
81 out = ((q31_t)(full_frac - frac) * value + (q31_t)value2 * frac) >> shift_size;
82 }
83 else
84 {
85 /* the largest positive value does not have a right side for linear interpolation */
86 out = value;
87 }
88
89 *pOut++ = out;
90 i--;
91 }
92}
