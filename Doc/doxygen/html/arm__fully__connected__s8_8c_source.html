50arm_status arm_fully_connected_s8(const cmsis_nn_context *ctx,
51 const cmsis_nn_fc_params *fc_params,
52 const cmsis_nn_per_tensor_quant_params *quant_params,
53 const cmsis_nn_dims *input_dims,
54 const q7_t *input,
55 const cmsis_nn_dims *filter_dims,
56 const q7_t *kernel,
57 const cmsis_nn_dims *bias_dims,
58 const int32_t *bias,
59 const cmsis_nn_dims *output_dims,
60 q7_t *output)
61{
62 (void)bias_dims;
63 (void)ctx;
64 (void)fc_params->filter_offset;
65
66 int32_t batch_cnt = input_dims->n;
67
68 while (batch_cnt)
69 {
70 arm_nn_vec_mat_mult_t_s8(input,
71 kernel,
72 bias,
73 output,
74 fc_params->input_offset,
75 0,
76 fc_params->output_offset,
77 quant_params->multiplier,
78 quant_params->shift,
79 filter_dims->n, /* col_dim or accum_depth */
80 output_dims->c, /* row_dim or output_depth */
81 fc_params->activation.min,
82 fc_params->activation.max,
83 1L);
84 input += filter_dims->n;
85 output += output_dims->c;
86 batch_cnt--;
87 }
88 return (ARM_MATH_SUCCESS);
89}
