301/* ----------------------------------------------------------------------
302 * Graphic equalizer Example
303 * ------------------------------------------------------------------- */
304
305int32_t main(void)
306{
307 float32_t *inputF32, *outputF32;
308 arm_biquad_cas_df1_32x64_ins_q31 S1;
309 arm_biquad_cas_df1_32x64_ins_q31 S2;
310 arm_biquad_casd_df1_inst_q31 S3;
311 arm_biquad_casd_df1_inst_q31 S4;
312 arm_biquad_casd_df1_inst_q31 S5;
313 int i;
314 int32_t status;
315
316 inputF32 = &testInput_f32[0];
317 outputF32 = &testOutput[0];
318
319 /* Initialize the state and coefficient buffers for all Biquad sections */
320
321 arm_biquad_cas_df1_32x64_init_q31(&S1, NUMSTAGES,
322 (q31_t *) &coeffTable[190*0 + 10*(gainDB[0] + 9)],
323 &biquadStateBand1Q31[0], 2);
324
325 arm_biquad_cas_df1_32x64_init_q31(&S2, NUMSTAGES,
326 (q31_t *) &coeffTable[190*1 + 10*(gainDB[1] + 9)],
327 &biquadStateBand2Q31[0], 2);
328
329 arm_biquad_cascade_df1_init_q31(&S3, NUMSTAGES,
330 (q31_t *) &coeffTable[190*2 + 10*(gainDB[2] + 9)],
331 &biquadStateBand3Q31[0], 2);
332
333 arm_biquad_cascade_df1_init_q31(&S4, NUMSTAGES,
334 (q31_t *) &coeffTable[190*3 + 10*(gainDB[3] + 9)],
335 &biquadStateBand4Q31[0], 2);
336
337 arm_biquad_cascade_df1_init_q31(&S5, NUMSTAGES,
338 (q31_t *) &coeffTable[190*4 + 10*(gainDB[4] + 9)],
339 &biquadStateBand5Q31[0], 2);
340
341
342 /* Call the process functions and needs to change filter coefficients
343 for varying the gain of each band */
344
345 for(i=0; i < NUMBLOCKS; i++)
346 {
347
348 /* ----------------------------------------------------------------------
349 ** Convert block of input data from float to Q31
350 ** ------------------------------------------------------------------- */
351
352 arm_float_to_q31(inputF32 + (i*BLOCKSIZE), inputQ31, BLOCKSIZE);
353
354 /* ----------------------------------------------------------------------
355 ** Scale down by 1/8. This provides additional headroom so that the
356 ** graphic EQ can apply gain.
357 ** ------------------------------------------------------------------- */
358
359 arm_scale_q31(inputQ31, 0x7FFFFFFF, -3, inputQ31, BLOCKSIZE);
360
361 /* ----------------------------------------------------------------------
362 ** Call the Q31 Biquad Cascade DF1 32x64 process function for band1, band2
363 ** ------------------------------------------------------------------- */
364
365 arm_biquad_cas_df1_32x64_q31(&S1, inputQ31, outputQ31, BLOCKSIZE);
366 arm_biquad_cas_df1_32x64_q31(&S2, outputQ31, outputQ31, BLOCKSIZE);
367
368 /* ----------------------------------------------------------------------
369 ** Call the Q31 Biquad Cascade DF1 process function for band3, band4, band5
370 ** ------------------------------------------------------------------- */
371
372 arm_biquad_cascade_df1_q31(&S3, outputQ31, outputQ31, BLOCKSIZE);
373 arm_biquad_cascade_df1_q31(&S4, outputQ31, outputQ31, BLOCKSIZE);
374 arm_biquad_cascade_df1_q31(&S5, outputQ31, outputQ31, BLOCKSIZE);
375
376 /* ----------------------------------------------------------------------
377 ** Convert Q31 result back to float
378 ** ------------------------------------------------------------------- */
379
380 arm_q31_to_float(outputQ31, outputF32 + (i * BLOCKSIZE), BLOCKSIZE);
381
382 /* ----------------------------------------------------------------------
383 ** Scale back up
384 ** ------------------------------------------------------------------- */
385
386 arm_scale_f32(outputF32 + (i * BLOCKSIZE), 8.0f, outputF32 + (i * BLOCKSIZE), BLOCKSIZE);
387 };
388
389 snr = arm_snr_f32(testRefOutput_f32, testOutput, TESTLENGTH);
390
391 status = (snr < SNR_THRESHOLD_F32) ? ARM_MATH_TEST_FAILURE : ARM_MATH_SUCCESS;
392
393 if (status != ARM_MATH_SUCCESS)
394 {
395#if defined (SEMIHOSTING)
396 printf("FAILURE\n");
397#else
398 while (1); /* main function does not return */
399#endif
400 }
401 else
402 {
403#if defined (SEMIHOSTING)
404 printf("SUCCESS\n");
405#else
406 while (1); /* main function does not return */
407#endif
408 }
409
410}
