43static void depthwise_conv_s8_mult_4(const int8_t *input,
44 const int32_t input_x,
45 const int32_t input_y,
46 const int32_t input_ch,
47 const int8_t *kernel,
48 const int32_t output_ch,
49 const int32_t ch_mult,
50 const int32_t kernel_x,
51 const int32_t kernel_y,
52 const int32_t pad_x,
53 const int32_t pad_y,
54 const int32_t stride_x,
55 const int32_t stride_y,
56 const int32_t *bias,
57 int8_t *output,
58 const int32_t *output_shift,
59 const int32_t *output_mult,
60 const int32_t output_x,
61 const int32_t output_y,
62 const int32_t output_offset,
63 const int32_t input_offset,
64 const int32_t output_activation_min,
65 const int32_t output_activation_max)
66{
67 for (int32_t in_h = -pad_y, out_h = 0, out_idx = 0; out_h < output_y; in_h += stride_y, ++out_h)
68 {
69 for (int32_t in_w = -pad_x, out_w = 0, ker_h_start = MAX(0, -in_h); out_w < output_x; in_w += stride_x, ++out_w)
70 {
71 for (int32_t in_ch = 0, out_ch = 0, ker_w_start = MAX(0, -in_w); out_ch < output_ch;
72 ++in_ch, out_ch += ch_mult)
73 {
74 for (int mult_tile = 0; mult_tile < ch_mult; mult_tile += 4)
75 {
76 int32_t out_buff[4] = {0, 0, 0, 0};
77 if (bias)
78 {
79 out_buff[0] = bias[out_ch + 0 + mult_tile];
80 out_buff[1] = bias[out_ch + 1 + mult_tile];
81 out_buff[2] = bias[out_ch + 2 + mult_tile];
82 out_buff[3] = bias[out_ch + 3 + mult_tile];
83 }
84
85 for (int32_t ker_h = ker_h_start; ker_h < MIN(kernel_y, input_y - in_h); ++ker_h)
86 {
87 int32_t ker_idx = ker_h * (output_ch * kernel_x) + ker_w_start * output_ch + out_ch;
88 int32_t in_idx = (in_h + ker_h) * (input_ch * input_x) + in_w * input_ch + in_ch;
89#if defined(__ARMCC_VERSION) && (__ARMCC_VERSION >= 6010050)
90#pragma clang loop unroll(disable)
91#endif
92 for (int32_t ker_w = ker_w_start; ker_w < MIN(kernel_x, input_x - in_w);
93 ++ker_w, ker_idx += output_ch)
94 {
95 int32_t in_val = input[in_idx + ker_w * input_ch] + input_offset;
96 out_buff[0] += in_val * kernel[ker_idx + 0 + mult_tile];
97 out_buff[1] += in_val * kernel[ker_idx + 1 + mult_tile];
98 out_buff[2] += in_val * kernel[ker_idx + 2 + mult_tile];
99 out_buff[3] += in_val * kernel[ker_idx + 3 + mult_tile];
100 }
101 }
102#if defined(ARM_MATH_MVEI)
103 (void)out_idx;
104 int32x4_t res = vldrwq_s32(out_buff);
105 res = arm_requantize_mve_32x4(res,
106 vldrwq_s32(&output_mult[out_ch + mult_tile]),
107 vldrwq_s32(&output_shift[out_ch + mult_tile]));
108 res = vaddq_n_s32(res, output_offset);
109
110 res = vmaxq_s32(res, vdupq_n_s32(output_activation_min));
111 res = vminq_s32(res, vdupq_n_s32(output_activation_max));
112 vstrbq_s32(output, res);
113 output += 4;
114#else
115 out_buff[0] = arm_nn_requantize(
116 out_buff[0], output_mult[out_ch + 0 + mult_tile], output_shift[out_ch + 0 + mult_tile]);
117 out_buff[1] = arm_nn_requantize(
118 out_buff[1], output_mult[out_ch + 1 + mult_tile], output_shift[out_ch + 1 + mult_tile]);
119 out_buff[2] = arm_nn_requantize(
120 out_buff[2], output_mult[out_ch + 2 + mult_tile], output_shift[out_ch + 2 + mult_tile]);
121 out_buff[3] = arm_nn_requantize(
122 out_buff[3], output_mult[out_ch + 3 + mult_tile], output_shift[out_ch + 3 + mult_tile]);
123
124 out_buff[0] += output_offset;
125 out_buff[1] += output_offset;
126 out_buff[2] += output_offset;
127 out_buff[3] += output_offset;
128
129 out_buff[0] = MIN(MAX(out_buff[0], output_activation_min), output_activation_max);
130 out_buff[1] = MIN(MAX(out_buff[1], output_activation_min), output_activation_max);
131 out_buff[2] = MIN(MAX(out_buff[2], output_activation_min), output_activation_max);
132 out_buff[3] = MIN(MAX(out_buff[3], output_activation_min), output_activation_max);
133
134 output[out_idx++] = (int8_t)out_buff[0];
135 output[out_idx++] = (int8_t)out_buff[1];
136 output[out_idx++] = (int8_t)out_buff[2];
137 output[out_idx++] = (int8_t)out_buff[3];
138
139#endif
140 }
141 }
142 }
143 }

146static void depthwise_conv_s8_generic(const q7_t *input,
147 const uint16_t input_batches,
148 const uint16_t input_x,
149 const uint16_t input_y,
150 const uint16_t input_ch,
151 const q7_t *kernel,
152 const uint16_t output_ch,
153 const uint16_t ch_mult,
154 const uint16_t kernel_x,
155 const uint16_t kernel_y,
156 const uint16_t pad_x,
157 const uint16_t pad_y,
158 const uint16_t stride_x,
159 const uint16_t stride_y,
160 const int32_t *bias,
161 q7_t *output,
162 const int32_t *output_shift,
163 const int32_t *output_mult,
164 const uint16_t output_x,
165 const uint16_t output_y,
166 const int32_t output_offset,
167 const int32_t input_offset,
168 const int32_t output_activation_min,
169 const int32_t output_activation_max,
170 const uint16_t dilation_x,
171 const uint16_t dilation_y)
172
173{
174 (void)output_ch;
175 int i_out = 0;
176 int i_batch;
177
178 for (i_batch = 0; i_batch < input_batches; i_batch++)
179 {
180 for (int i_out_y = 0; i_out_y < output_y; i_out_y++)
181 {
182 const int16_t base_idx_y = (i_out_y * stride_y) - pad_y;
183 for (int i_out_x = 0; i_out_x < output_x; i_out_x++)
184 {
185 const int16_t base_idx_x = (i_out_x * stride_x) - pad_x;
186 for (int i_input_ch = 0; i_input_ch < input_ch; i_input_ch++)
187 {
188 for (int i_ch_mult = 0; i_ch_mult < ch_mult; i_ch_mult++)
189 {
190 const int idx_out_ch = i_ch_mult + i_input_ch * ch_mult;
191 int32_t acc_0 = 0;
192
193 int ker_y_start;
194 int ker_x_start;
195 int ker_y_end;
196 int ker_x_end;
197
198 if (dilation_x > 1)
199 {
200 const int32_t start_x_max = (-base_idx_x + dilation_x - 1) / dilation_x;
201 ker_x_start = MAX(0, start_x_max);
202 const int32_t end_min_x = (input_x - base_idx_x + dilation_x - 1) / dilation_x;
203 ker_x_end = MIN(kernel_x, end_min_x);
204 }
205 else
206 {
207 ker_x_start = MAX(0, -base_idx_x);
208 ker_x_end = MIN(kernel_x, input_x - base_idx_x);
209 }
210
211 if (dilation_y > 1)
212 {
213 const int32_t start_y_max = (-base_idx_y + dilation_y - 1) / dilation_y;
214 ker_y_start = MAX(0, start_y_max);
215 const int32_t end_min_y = (input_y - base_idx_y + dilation_y - 1) / dilation_y;
216 ker_y_end = MIN(kernel_y, end_min_y);
217 }
218 else
219 {
220 ker_y_start = MAX(0, -base_idx_y);
221 ker_y_end = MIN(kernel_y, input_y - base_idx_y);
222 }
223
224 if (bias)
225 {
226 acc_0 = bias[idx_out_ch];
227 }
228
229 for (int i_ker_y = ker_y_start; i_ker_y < ker_y_end; i_ker_y++)
230 {
231 const int32_t idx_y = base_idx_y + dilation_y * i_ker_y;
232 for (int i_ker_x = ker_x_start; i_ker_x < ker_x_end; i_ker_x++)
233 {
234 const int32_t idx_x = base_idx_x + dilation_x * i_ker_x;
235 int32_t idx_0 = (idx_y * input_x + idx_x) * input_ch + i_input_ch;
236 int32_t ker_idx_0 = (i_ker_y * kernel_x + i_ker_x) * (input_ch * ch_mult) + idx_out_ch;
237
238 acc_0 += (input[idx_0] + input_offset) * kernel[ker_idx_0];
239 }
240 }
241
242 /* Requantize and clamp output to provided range */
243 acc_0 = arm_nn_requantize(acc_0, output_mult[idx_out_ch], output_shift[idx_out_ch]);
244 acc_0 += output_offset;
245 acc_0 = MAX(acc_0, output_activation_min);
246 acc_0 = MIN(acc_0, output_activation_max);
247
248 output[i_out++] = acc_0;
249 }
250 }
251 }
252 }
253 /* Advance to the next batch */
254 input += (input_x * input_y * input_ch);
255 }
256}

258/*
259 * Basic s8 depthwise convolution function.
260 *
261 * Refer header file for details.
262 * Optimization using DSP extension is not available for the generic case where channel multiplier is > 1.
263 *
264 */
265arm_status arm_depthwise_conv_s8(const cmsis_nn_context *ctx,
266 const cmsis_nn_dw_conv_params *dw_conv_params,
267 const cmsis_nn_per_channel_quant_params *quant_params,
268 const cmsis_nn_dims *input_dims,
269 const q7_t *input,
270 const cmsis_nn_dims *filter_dims,
271 const q7_t *kernel,
272 const cmsis_nn_dims *bias_dims,
273 const int32_t *bias,
274 const cmsis_nn_dims *output_dims,
275 q7_t *output)
276{
277 const uint16_t dilation_x = dw_conv_params->dilation.w;
278 const uint16_t dilation_y = dw_conv_params->dilation.h;
279
280 (void)dw_conv_params->dilation;
281 (void)bias_dims;
282 (void)ctx;
283
284 if (dw_conv_params->ch_mult % 4 == 0 && input_dims->n == 1 && dw_conv_params->dilation.w == 1 &&
285 dw_conv_params->dilation.h == 1)
286 {
287 depthwise_conv_s8_mult_4(input,
288 input_dims->w,
289 input_dims->h,
290 input_dims->c,
291 kernel,
292 output_dims->c,
293 dw_conv_params->ch_mult,
294 filter_dims->w,
295 filter_dims->h,
296 dw_conv_params->padding.w,
297 dw_conv_params->padding.h,
298 dw_conv_params->stride.w,
299 dw_conv_params->stride.h,
300 bias,
301 output,
302 quant_params->shift,
303 quant_params->multiplier,
304 output_dims->w,
305 output_dims->h,
306 dw_conv_params->output_offset,
307 dw_conv_params->input_offset,
308 dw_conv_params->activation.min,
309 dw_conv_params->activation.max);
310 }
311 else
312 {
313 depthwise_conv_s8_generic(input,
314 input_dims->n,
315 input_dims->w,
316 input_dims->h,
317 input_dims->c,
318 kernel,
319 output_dims->c,
320 dw_conv_params->ch_mult,
321 filter_dims->w,
322 filter_dims->h,
323 dw_conv_params->padding.w,
324 dw_conv_params->padding.h,
325 dw_conv_params->stride.w,
326 dw_conv_params->stride.h,
327 bias,
328 output,
329 quant_params->shift,
330 quant_params->multiplier,
331 output_dims->w,
332 output_dims->h,
333 dw_conv_params->output_offset,
334 dw_conv_params->input_offset,
335 dw_conv_params->activation.min,
336 dw_conv_params->activation.max,
337 dilation_x,
338 dilation_y);
339 }
340
341 /* Return to application */
342 return ARM_MATH_SUCCESS;
343}
