205void ADC_Task(void *pvParameters)
206{
207 ADC_Driver_t *driver = (ADC_Driver_t *)pvParameters;
208 ADC_Data_t adc_data;
209
210 for (;;) {
211 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
212
213 if (xSemaphoreTake(driver->data_mutex, pdMS_TO_TICKS(100)) == pdPASS) {
214 adc_data.channel_count = driver->channel_count;
215 adc_data.timestamp = xTaskGetTickCount();
216
217 for (int i = 0; i < ADC_CHANNEL_COUNT; i++) {
218 adc_data.channel_values[i] = driver->raw_buffer[i];
219 }
220
221 if (uxQueueSpacesAvailable(driver->data_queue) == 0) {
222 ADC_Data_t dummy;
223 xQueueReceive(driver->data_queue, &dummy, 0);
224 }
225
226 xQueueSend(driver->data_queue, &adc_data, 0);
227 xSemaphoreGive(driver->data_mutex);
228 }
229 }
230}
