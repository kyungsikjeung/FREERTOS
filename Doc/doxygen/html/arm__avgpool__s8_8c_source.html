36static void scale_q31_to_q7_and_clamp(const q31_t *buffer,
37 q7_t *target,
38 int32_t length,
39 const int32_t count,
40 const int act_min,
41 const int act_max)
42{
43 const int half_count = count / 2;
44
45 // Prevent static code issue DIVIDE_BY_ZERO.
46 if (count == 0)
47 {
48 return;
49 }
50
51 for (int i = 0; i < length; i++)
52 {
53 int32_t sum = buffer[i] > 0 ? (buffer[i] + half_count) : (buffer[i] - half_count);
54 sum = sum / count;
55 sum = MAX(sum, act_min);
56 sum = MIN(sum, act_max);
57
58 target[i] = (q7_t)sum;
59 }
60}

81arm_status arm_avgpool_s8(const cmsis_nn_context *ctx,
82 const cmsis_nn_pool_params *pool_params,
83 const cmsis_nn_dims *input_dims,
84 const q7_t *src,
85 const cmsis_nn_dims *filter_dims,
86 const cmsis_nn_dims *output_dims,
87 q7_t *dst)
88{
89 (void)ctx;
90 const int32_t input_y = input_dims->h;
91 const int32_t input_x = input_dims->w;
92 const int32_t output_y = output_dims->h;
93 const int32_t output_x = output_dims->w;
94 const int32_t stride_y = pool_params->stride.h;
95 const int32_t stride_x = pool_params->stride.w;
96 const int32_t kernel_y = filter_dims->h;
97 const int32_t kernel_x = filter_dims->w;
98 const int32_t pad_y = pool_params->padding.h;
99 const int32_t pad_x = pool_params->padding.w;
100 const int32_t act_min = pool_params->activation.min;
101 const int32_t act_max = pool_params->activation.max;
102 const int32_t ch_src = input_dims->c;
103
104 int32_t i_x, i_y;
105 int32_t k_x, k_y;
106
107 for (i_y = 0; i_y < output_y; i_y++)
108 {
109 for (i_x = 0; i_x < output_x; i_x++)
110 {
111
112 int32_t k_y_start, k_y_end;
113 int32_t k_x_start, k_x_end;
114 int32_t chCnt;
115 const int8_t *pTmp, *pTmpInner;
116 int8_t *pDst;
117
118 k_y_start = MAX(0, i_y * stride_y - pad_y);
119 k_y_end = MIN(i_y * stride_y - pad_y + kernel_y, input_y);
120
121 k_x_start = MAX(0, i_x * stride_x - pad_x);
122 k_x_end = MIN(i_x * stride_x - pad_x + kernel_x, input_x);
123
124 pTmp = src;
125 pDst = &dst[ch_src * (i_x + i_y * output_x)];
126
127 chCnt = ch_src >> 4;
128 while (chCnt > 0)
129 {
130 int32x4_t sumV1, sumV2, sumV3, sumV4;
131
132 int8x16_t tempV;
133 int16x8_t tempVLO, tempVHI;
134 int32x4_t tempVLOLO, tempVLOHI, tempVHILO, tempVHIHI;
135 int32_t count = 0;
136
137 sumV1 = vdupq_n_s32(0);
138 sumV2 = vdupq_n_s32(0);
139 sumV3 = vdupq_n_s32(0);
140 sumV4 = vdupq_n_s32(0);
141
142 for (k_y = k_y_start; k_y < k_y_end; k_y++)
143 {
144 for (k_x = k_x_start; k_x < k_x_end; k_x++)
145 {
146 pTmpInner = pTmp + (ch_src * (k_x + k_y * input_x));
147 tempV = vldrbq_s8(pTmpInner);
148
149 tempVLO = vmovlbq_s8(tempV);
150 tempVHI = vmovltq_s8(tempV);
151
152 tempVLOLO = vmovlbq_s16(tempVLO);
153 tempVLOHI = vmovltq_s16(tempVLO);
154
155 tempVHILO = vmovlbq_s16(tempVHI);
156 tempVHIHI = vmovltq_s16(tempVHI);
157
158 sumV1 = vaddq_s32(sumV1, tempVLOLO);
159 sumV2 = vaddq_s32(sumV2, tempVLOHI);
160 sumV3 = vaddq_s32(sumV3, tempVHILO);
161 sumV4 = vaddq_s32(sumV4, tempVHIHI);
162
163 count++;
164 }
165 }
166
167 // Prevent static code issue DIVIDE_BY_ZERO.
168 if (count == 0)
169 {
170 return ARM_MATH_ARGUMENT_ERROR;
171 }
172
173 sumV1[0] = sumV1[0] > 0 ? (sumV1[0] + count / 2) / count : (sumV1[0] - count / 2) / count;
174 sumV1[1] = sumV1[1] > 0 ? (sumV1[1] + count / 2) / count : (sumV1[1] - count / 2) / count;
175 sumV1[2] = sumV1[2] > 0 ? (sumV1[2] + count / 2) / count : (sumV1[2] - count / 2) / count;
176 sumV1[3] = sumV1[3] > 0 ? (sumV1[3] + count / 2) / count : (sumV1[3] - count / 2) / count;
177
178 sumV2[0] = sumV2[0] > 0 ? (sumV2[0] + count / 2) / count : (sumV2[0] - count / 2) / count;
179 sumV2[1] = sumV2[1] > 0 ? (sumV2[1] + count / 2) / count : (sumV2[1] - count / 2) / count;
180 sumV2[2] = sumV2[2] > 0 ? (sumV2[2] + count / 2) / count : (sumV2[2] - count / 2) / count;
181 sumV2[3] = sumV2[3] > 0 ? (sumV2[3] + count / 2) / count : (sumV2[3] - count / 2) / count;
182
183 sumV3[0] = sumV3[0] > 0 ? (sumV3[0] + count / 2) / count : (sumV3[0] - count / 2) / count;
184 sumV3[1] = sumV3[1] > 0 ? (sumV3[1] + count / 2) / count : (sumV3[1] - count / 2) / count;
185 sumV3[2] = sumV3[2] > 0 ? (sumV3[2] + count / 2) / count : (sumV3[2] - count / 2) / count;
186 sumV3[3] = sumV3[3] > 0 ? (sumV3[3] + count / 2) / count : (sumV3[3] - count / 2) / count;
187
188 sumV4[0] = sumV4[0] > 0 ? (sumV4[0] + count / 2) / count : (sumV4[0] - count / 2) / count;
189 sumV4[1] = sumV4[1] > 0 ? (sumV4[1] + count / 2) / count : (sumV4[1] - count / 2) / count;
190 sumV4[2] = sumV4[2] > 0 ? (sumV4[2] + count / 2) / count : (sumV4[2] - count / 2) / count;
191 sumV4[3] = sumV4[3] > 0 ? (sumV4[3] + count / 2) / count : (sumV4[3] - count / 2) / count;
192
193 sumV1 = vmaxq_s32(sumV1, vdupq_n_s32(act_min));
194 sumV1 = vminq_s32(sumV1, vdupq_n_s32(act_max));
195
196 sumV2 = vmaxq_s32(sumV2, vdupq_n_s32(act_min));
197 sumV2 = vminq_s32(sumV2, vdupq_n_s32(act_max));
198
199 sumV3 = vmaxq_s32(sumV3, vdupq_n_s32(act_min));
200 sumV3 = vminq_s32(sumV3, vdupq_n_s32(act_max));
201
202 sumV4 = vmaxq_s32(sumV4, vdupq_n_s32(act_min));
203 sumV4 = vminq_s32(sumV4, vdupq_n_s32(act_max));
204
205 tempVLO = vmovnbq_s32(tempVLO, sumV1);
206 tempVLO = vmovntq_s32(tempVLO, sumV2);
207
208 tempVHI = vmovnbq_s32(tempVHI, sumV3);
209 tempVHI = vmovntq_s32(tempVHI, sumV4);
210
211 tempV = vmovnbq_s16(tempV, tempVLO);
212 tempV = vmovntq_s16(tempV, tempVHI);
213
214 vstrbq_s8(pDst, tempV);
215 pDst += 16;
216
217 chCnt--;
218 pTmp += 16;
219 }
220
221 chCnt = ch_src & 0xF;
222 while (chCnt > 0)
223 {
224 int32_t sum = 0;
225 int32_t count = 0;
226
227 for (k_y = k_y_start; k_y < k_y_end; k_y++)
228 {
229 for (k_x = k_x_start; k_x < k_x_end; k_x++)
230 {
231 sum += pTmp[ch_src * (k_x + k_y * input_x)];
232 count++;
233 }
234 }
235
236 // Prevent static code issue DIVIDE_BY_ZERO.
237 if (count == 0)
238 {
239 return ARM_MATH_ARGUMENT_ERROR;
240 }
241
242 sum = sum > 0 ? (sum + count / 2) / count : (sum - count / 2) / count;
243 sum = MAX(sum, act_min);
244 sum = MIN(sum, act_max);
245
246 *pDst++ = sum;
247
248 chCnt--;
249 pTmp++;
250 }
251 }
252 }
253 return ARM_MATH_SUCCESS;
254}

257arm_status arm_avgpool_s8(const cmsis_nn_context *ctx,
258 const cmsis_nn_pool_params *pool_params,
259 const cmsis_nn_dims *input_dims,
260 const q7_t *src,
261 const cmsis_nn_dims *filter_dims,
262 const cmsis_nn_dims *output_dims,
263 q7_t *dst)
264{
265 const int32_t input_y = input_dims->h;
266 const int32_t input_x = input_dims->w;
267 const int32_t output_y = output_dims->h;
268 const int32_t output_x = output_dims->w;
269 const int32_t stride_y = pool_params->stride.h;
270 const int32_t stride_x = pool_params->stride.w;
271 const int32_t kernel_y = filter_dims->h;
272 const int32_t kernel_x = filter_dims->w;
273 const int32_t pad_y = pool_params->padding.h;
274 const int32_t pad_x = pool_params->padding.w;
275 const int32_t act_min = pool_params->activation.min;
276 const int32_t act_max = pool_params->activation.max;
277 const int32_t ch_src = input_dims->c;
278
279 if (ctx->buf == NULL && arm_avgpool_s8_get_buffer_size(output_dims->w, input_dims->c))
280 {
281 return ARM_MATH_ARGUMENT_ERROR;
282 }
283 q31_t *buffer = (q31_t *)ctx->buf;
284
285#if defined(ARM_MATH_DSP)
286
287 /* Run the following code for CPU's with DSP extension
288 */
289 for (int i_y = 0, idx_y = -pad_y; i_y < output_y; idx_y += stride_y, i_y++)
290 {
291 for (int i_x = 0, idx_x = -pad_x; i_x < output_x; idx_x += stride_x, i_x++)
292 {
293 /* Condition for kernel start dimension:
294 (base_idx_<x,y> + kernel_<x,y>_start) >= 0 */
295 const int32_t kernel_y_start = MAX(0, -idx_y);
296 const int32_t kernel_x_start = MAX(0, -idx_x);
297
298 /* Condition for kernel end dimension:
299 (base_idx_<x,y> + kernel_<x,y>_end) < dim_src_<width,height> */
300 const int32_t kernel_y_end = MIN(kernel_y, input_y - idx_y);
301 const int32_t kernel_x_end = MIN(kernel_x, input_x - idx_x);
302
303 int count = 0;
304
305 for (int k_y = kernel_y_start; k_y < kernel_y_end; k_y++)
306 {
307 for (int k_x = kernel_x_start; k_x < kernel_x_end; k_x++)
308 {
309 const q7_t *start = src + ch_src * (k_x + idx_x + (k_y + idx_y) * input_x);
310
311 if (count == 0)
312 {
313 for (int i = 0; i < ch_src; i++)
314 {
315 buffer[i] = start[i];
316 }
317 }
318 else
319 {
320 for (int i = 0; i < ch_src; i++)
321 {
322 buffer[i] = __QADD(start[i], buffer[i]);
323 }
324 }
325 count++;
326 }
327 }
328
329 // Prevent static code issue DIVIDE_BY_ZERO.
330 if (count == 0)
331 {
332 return ARM_MATH_ARGUMENT_ERROR;
333 }
334
335 scale_q31_to_q7_and_clamp(buffer, dst, ch_src, count, act_min, act_max);
336 dst += ch_src;
337 }
338 }
339#else
340
341 /* Reference C code adapted from CMSIS-NN arm_avepool_q7_HWC.
342 */
343 (void)buffer;
344 int16_t i_ch_in, i_x, i_y;
345 int16_t k_x, k_y;
346
347 for (i_y = 0; i_y < output_y; i_y++)
348 {
349 for (i_x = 0; i_x < output_x; i_x++)
350 {
351 for (i_ch_in = 0; i_ch_in < ch_src; i_ch_in++)
352 {
353 int sum = 0;
354 int count = 0;
355 for (k_y = i_y * stride_y - pad_y; k_y < i_y * stride_y - pad_y + kernel_y; k_y++)
356 {
357 for (k_x = i_x * stride_x - pad_x; k_x < i_x * stride_x - pad_x + kernel_x; k_x++)
358 {
359 if (k_y >= 0 && k_x >= 0 && k_y < input_y && k_x < input_x)
360 {
361 sum += src[i_ch_in + ch_src * (k_x + k_y * input_x)];
362 count++;
363 }
364 }
365 }
366
367 // Prevent static code issue DIVIDE_BY_ZERO.
368 if (count == 0)
369 {
370 return ARM_MATH_ARGUMENT_ERROR;
371 }
372
373 sum = sum > 0 ? (sum + count / 2) / count : (sum - count / 2) / count;
374 sum = MAX(sum, act_min);
375 sum = MIN(sum, act_max);
376
377 dst[i_ch_in + ch_src * (i_x + i_y * output_x)] = sum;
378 }
379 }
380 }
381
382#endif
383 return ARM_MATH_SUCCESS;
384}
