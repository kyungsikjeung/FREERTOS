43/*
44 * Softmax function with s8 input and output of s8 or s16.
45 *
46 * Refer header file for details.
47 *
48 */
49void arm_nn_softmax_common_s8(const int8_t *input,
50 const int32_t num_rows,
51 const int32_t row_size,
52 const int32_t mult,
53 const int32_t shift,
54 const int32_t diff_min,
55 const bool int16_output,
56 void *output)
57{
58 const int32_t mask = (1 << shift);
59
60 int32_t col = 0;
61 int32_t row_idx;
62
63 for (row_idx = 0; row_idx < num_rows; ++row_idx)
64 {
65 // Find the maximum value in order to ensure numerical stability
66 int8_t max = *input;
67
68 for (col = 1; col < row_size; ++col)
69 {
70 max = MAX(max, input[col]);
71 }
72
73 int32_t diff = 0;
74 int32_t sum = 0;
75
76 for (col = 0; col < row_size; ++col)
77 {
78 diff = input[col] - max;
79 if (diff >= diff_min)
80 {
81 sum += DIV_POW2(EXP_ON_NEG(MUL_SAT(diff * mask, mult)), ACCUM_BITS);
82 }
83 }
84
85 const int32_t headroom = __CLZ(sum);
86 const int32_t shifted_scale = ONE_OVER1((sum > 0 ? sum << headroom : 0) - (1 << 31));
87 int32_t bits_over_unit;
88
89 if (int16_output)
90 {
91 int16_t *output_s16 = (int16_t *)output + row_idx * row_size;
92
93 bits_over_unit = ACCUM_BITS - headroom + 15;
94
95 for (col = 0; col < row_size; ++col)
96 {
97 diff = input[col] - max;
98
99 if (diff >= diff_min)
100 {
101 const int32_t res =
102 DIV_POW2(MUL_SAT(shifted_scale, EXP_ON_NEG(MUL_SAT(diff * mask, mult))), bits_over_unit) +
103 NN_Q15_MIN;
104 output_s16[col] = (int16_t)CLAMP(res, (int32_t)NN_Q15_MAX, (int32_t)NN_Q15_MIN);
105 }
106 else
107 {
108 output_s16[col] = NN_Q15_MIN;
109 }
110 }
111 }
112 else
113 {
114 int8_t *output_s8 = (int8_t *)output + row_idx * row_size;
115
116 bits_over_unit = ACCUM_BITS - headroom + 23;
117
118 for (col = 0; col < row_size; ++col)
119 {
120 diff = input[col] - max;
121 if (diff >= diff_min)
122 {
123 const int32_t res =
124 DIV_POW2(MUL_SAT(shifted_scale, EXP_ON_NEG(MUL_SAT(diff * mask, mult))), bits_over_unit) +
125 NN_Q7_MIN;
126 output_s8[col] = (int8_t)CLAMP(res, (int32_t)NN_Q7_MAX, (int32_t)NN_Q7_MIN);
127 }
128 else
129 {
130 output_s8[col] = NN_Q7_MIN;
131 }
132 }
133 }
134
135 input += row_size;
136 }
137}
