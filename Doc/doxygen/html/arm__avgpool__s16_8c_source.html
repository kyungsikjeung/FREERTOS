44 * s16 average pooling function
45 *
46 * Refer to header file for details.
47 *
48 */
49arm_status arm_avgpool_s16(const cmsis_nn_context *ctx,
50 const cmsis_nn_pool_params *pool_params,
51 const cmsis_nn_dims *input_dims,
52 const q15_t *src,
53 const cmsis_nn_dims *filter_dims,
54 const cmsis_nn_dims *output_dims,
55 q15_t *dst)
56{
57 (void)ctx;
58 const int32_t input_y = input_dims->h;
59 const int32_t input_x = input_dims->w;
60 const int32_t output_y = output_dims->h;
61 const int32_t output_x = output_dims->w;
62 const int32_t stride_y = pool_params->stride.h;
63 const int32_t stride_x = pool_params->stride.w;
64 const int32_t kernel_y = filter_dims->h;
65 const int32_t kernel_x = filter_dims->w;
66 const int32_t pad_y = pool_params->padding.h;
67 const int32_t pad_x = pool_params->padding.w;
68 const int32_t act_min = pool_params->activation.min;
69 const int32_t act_max = pool_params->activation.max;
70 const int32_t ch_src = input_dims->c;
71
72 /* Reference C code adapted from CMSIS-NN arm_avgpool_s8.c.
73 */
74
75 for (int i_y = 0, base_idx_y = -pad_y; i_y < output_y; base_idx_y += stride_y, i_y++)
76 {
77 for (int i_x = 0, base_idx_x = -pad_x; i_x < output_x; base_idx_x += stride_x, i_x++)
78 {
79 /* Condition for kernel start dimension: (base_idx_<x,y> + kernel_<x,y>_start) >= 0 */
80 const int32_t ker_y_start = MAX(0, -base_idx_y);
81 const int32_t ker_x_start = MAX(0, -base_idx_x);
82
83 /* Condition for kernel end dimension: (base_idx_<x,y> + kernel_<x,y>_end) < dim_src_<width,height> */
84 const int32_t kernel_y_end = MIN(kernel_y, input_y - base_idx_y);
85 const int32_t kernel_x_end = MIN(kernel_x, input_x - base_idx_x);
86
87 for (int i_ch_in = 0; i_ch_in < ch_src; i_ch_in++)
88 {
89 int sum = 0;
90 int count = 0;
91
92 for (int k_y = ker_y_start; k_y < kernel_y_end; k_y++)
93 {
94 for (int k_x = ker_x_start; k_x < kernel_x_end; k_x++)
95 {
96 sum += src[i_ch_in + ch_src * (k_x + base_idx_x + (k_y + base_idx_y) * input_x)];
97 count++;
98 }
99 }
100
101 // Prevent static code issue DIVIDE_BY_ZERO.
102 if (count == 0)
103 {
104 return ARM_MATH_ARGUMENT_ERROR;
105 }
106
107 sum = sum > 0 ? (sum + count / 2) / count : (sum - count / 2) / count;
108 sum = MAX(sum, act_min);
109 sum = MIN(sum, act_max);
110
111 dst[i_ch_in + ch_src * (i_x + i_y * output_x)] = sum;
112 }
113 }
114 }
115
116 return ARM_MATH_SUCCESS;
117}
