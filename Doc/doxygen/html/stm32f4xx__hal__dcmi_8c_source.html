177__weak HAL_StatusTypeDef HAL_DCMI_Init(DCMI_HandleTypeDef *hdcmi)
178{
179 /* Check the DCMI peripheral state */
180 if(hdcmi == NULL)
181 {
182 return HAL_ERROR;
183 }
184
185 /* Check function parameters */
186 assert_param(IS_DCMI_ALL_INSTANCE(hdcmi->Instance));
187 assert_param(IS_DCMI_PCKPOLARITY(hdcmi->Init.PCKPolarity));
188 assert_param(IS_DCMI_VSPOLARITY(hdcmi->Init.VSPolarity));
189 assert_param(IS_DCMI_HSPOLARITY(hdcmi->Init.HSPolarity));
190 assert_param(IS_DCMI_SYNCHRO(hdcmi->Init.SynchroMode));
191 assert_param(IS_DCMI_CAPTURE_RATE(hdcmi->Init.CaptureRate));
192 assert_param(IS_DCMI_EXTENDED_DATA(hdcmi->Init.ExtendedDataMode));
193 assert_param(IS_DCMI_MODE_JPEG(hdcmi->Init.JPEGMode));
194
195 if(hdcmi->State == HAL_DCMI_STATE_RESET)
196 {
197 /* Allocate lock resource and initialize it */
198 hdcmi->Lock = HAL_UNLOCKED;
199 /* Init the low level hardware */
200 /* Init the DCMI Callback settings */
201#if (USE_HAL_DCMI_REGISTER_CALLBACKS == 1)
202 hdcmi->FrameEventCallback = HAL_DCMI_FrameEventCallback; /* Legacy weak FrameEventCallback */
203 hdcmi->VsyncEventCallback = HAL_DCMI_VsyncEventCallback; /* Legacy weak VsyncEventCallback */
204 hdcmi->LineEventCallback = HAL_DCMI_LineEventCallback; /* Legacy weak LineEventCallback */
205 hdcmi->ErrorCallback = HAL_DCMI_ErrorCallback; /* Legacy weak ErrorCallback */
206
207 if(hdcmi->MspInitCallback == NULL)
208 {
209 /* Legacy weak MspInit Callback */
210 hdcmi->MspInitCallback = HAL_DCMI_MspInit;
211 }
212 /* Initialize the low level hardware (MSP) */
213 hdcmi->MspInitCallback(hdcmi);
214#else
215 /* Init the low level hardware : GPIO, CLOCK, NVIC and DMA */
216 HAL_DCMI_MspInit(hdcmi);
217#endif /* (USE_HAL_DCMI_REGISTER_CALLBACKS) */
218 HAL_DCMI_MspInit(hdcmi);
219 }
220
221 /* Change the DCMI state */
222 hdcmi->State = HAL_DCMI_STATE_BUSY;
223
224 /* Set DCMI parameters */
225 /* Configures the HS, VS, DE and PC polarity */
226 hdcmi->Instance->CR &= ~(DCMI_CR_PCKPOL | DCMI_CR_HSPOL | DCMI_CR_VSPOL | DCMI_CR_EDM_0 |
227 DCMI_CR_EDM_1 | DCMI_CR_FCRC_0 | DCMI_CR_FCRC_1 | DCMI_CR_JPEG |
228 DCMI_CR_ESS);
229 hdcmi->Instance->CR |= (uint32_t)(hdcmi->Init.SynchroMode | hdcmi->Init.CaptureRate | \
230 hdcmi->Init.VSPolarity | hdcmi->Init.HSPolarity | \
231 hdcmi->Init.PCKPolarity | hdcmi->Init.ExtendedDataMode | \
232 hdcmi->Init.JPEGMode);
233
234 if(hdcmi->Init.SynchroMode == DCMI_SYNCHRO_EMBEDDED)
235 {
236 hdcmi->Instance->ESCR = (((uint32_t)hdcmi->Init.SyncroCode.FrameStartCode) |
237 ((uint32_t)hdcmi->Init.SyncroCode.LineStartCode << DCMI_POSITION_ESCR_LSC)|
238 ((uint32_t)hdcmi->Init.SyncroCode.LineEndCode << DCMI_POSITION_ESCR_LEC) |
239 ((uint32_t)hdcmi->Init.SyncroCode.FrameEndCode << DCMI_POSITION_ESCR_FEC));
240 }
241
242 /* Enable the Line, Vsync, Error and Overrun interrupts */
243 __HAL_DCMI_ENABLE_IT(hdcmi, DCMI_IT_LINE | DCMI_IT_VSYNC | DCMI_IT_ERR | DCMI_IT_OVR);
244
245 /* Update error code */
246 hdcmi->ErrorCode = HAL_DCMI_ERROR_NONE;
247
248 /* Initialize the DCMI state*/
249 hdcmi->State = HAL_DCMI_STATE_READY;
250
251 return HAL_OK;
252}

347HAL_StatusTypeDef HAL_DCMI_Start_DMA(DCMI_HandleTypeDef* hdcmi, uint32_t DCMI_Mode, uint32_t pData, uint32_t Length)
348{
349 HAL_StatusTypeDef status;
350
351 /* Initialize the second memory address */
352 uint32_t SecondMemAddress = 0U;
353
354 /* Check function parameters */
355 assert_param(IS_DCMI_CAPTURE_MODE(DCMI_Mode));
356
357 /* Process Locked */
358 __HAL_LOCK(hdcmi);
359
360 /* Lock the DCMI peripheral state */
361 hdcmi->State = HAL_DCMI_STATE_BUSY;
362
363 /* Enable DCMI by setting DCMIEN bit */
364 __HAL_DCMI_ENABLE(hdcmi);
365
366 /* Configure the DCMI Mode */
367 hdcmi->Instance->CR &= ~(DCMI_CR_CM);
368 hdcmi->Instance->CR |= (uint32_t)(DCMI_Mode);
369
370 /* Set the DMA memory0 conversion complete callback */
371 hdcmi->DMA_Handle->XferCpltCallback = DCMI_DMAXferCplt;
372
373 /* Set the DMA error callback */
374 hdcmi->DMA_Handle->XferErrorCallback = DCMI_DMAError;
375
376 /* Set the dma abort callback */
377 hdcmi->DMA_Handle->XferAbortCallback = NULL;
378
379 /* Reset transfer counters value */
380 hdcmi->XferCount = 0U;
381 hdcmi->XferTransferNumber = 0U;
382
383 if(Length <= 0xFFFFU)
384 {
385 /* Enable the DMA Stream */
386 status = HAL_DMA_Start_IT(hdcmi->DMA_Handle, (uint32_t)&hdcmi->Instance->DR, (uint32_t)pData, Length);
387 }
388 else /* DCMI_DOUBLE_BUFFER Mode */
389 {
390 /* Set the DMA memory1 conversion complete callback */
391 hdcmi->DMA_Handle->XferM1CpltCallback = DCMI_DMAXferCplt;
392
393 /* Initialize transfer parameters */
394 hdcmi->XferCount = 1U;
395 hdcmi->XferSize = Length;
396 hdcmi->pBuffPtr = pData;
397
398 /* Get the number of buffer */
399 while(hdcmi->XferSize > 0xFFFFU)
400 {
401 hdcmi->XferSize = (hdcmi->XferSize/2U);
402 hdcmi->XferCount = hdcmi->XferCount*2U;
403 }
404
405 /* Update DCMI counter and transfer number*/
406 hdcmi->XferCount = (hdcmi->XferCount - 2U);
407 hdcmi->XferTransferNumber = hdcmi->XferCount;
408
409 /* Update second memory address */
410 SecondMemAddress = (uint32_t)(pData + (4U*hdcmi->XferSize));
411
412 /* Start DMA multi buffer transfer */
413 status = HAL_DMAEx_MultiBufferStart_IT(hdcmi->DMA_Handle, (uint32_t)&hdcmi->Instance->DR, (uint32_t)pData, SecondMemAddress, hdcmi->XferSize);
414 }
415
416 /* Enable Capture */
417 hdcmi->Instance->CR |= DCMI_CR_CAPTURE;
418
419 /* Release Lock */
420 __HAL_UNLOCK(hdcmi);
421
422 /* Return function status */
423 return status;
424}

432HAL_StatusTypeDef HAL_DCMI_Stop(DCMI_HandleTypeDef* hdcmi)
433{
434 __IO uint32_t count = SystemCoreClock / HAL_TIMEOUT_DCMI_STOP;
435 HAL_StatusTypeDef status = HAL_OK;
436
437 /* Process locked */
438 __HAL_LOCK(hdcmi);
439
440 /* Lock the DCMI peripheral state */
441 hdcmi->State = HAL_DCMI_STATE_BUSY;
442
443 /* Disable Capture */
444 hdcmi->Instance->CR &= ~(DCMI_CR_CAPTURE);
445
446 /* Check if the DCMI capture effectively disabled */
447 do
448 {
449 if (count-- == 0U)
450 {
451 /* Update error code */
452 hdcmi->ErrorCode |= HAL_DCMI_ERROR_TIMEOUT;
453
454 status = HAL_TIMEOUT;
455 break;
456 }
457 }
458 while((hdcmi->Instance->CR & DCMI_CR_CAPTURE) != 0U);
459
460 /* Disable the DCMI */
461 __HAL_DCMI_DISABLE(hdcmi);
462
463 /* Disable the DMA */
464 HAL_DMA_Abort(hdcmi->DMA_Handle);
465
466 /* Update error code */
467 hdcmi->ErrorCode |= HAL_DCMI_ERROR_NONE;
468
469 /* Change DCMI state */
470 hdcmi->State = HAL_DCMI_STATE_READY;
471
472 /* Process Unlocked */
473 __HAL_UNLOCK(hdcmi);
474
475 /* Return function status */
476 return status;
477}

488 HAL_StatusTypeDef status = HAL_OK;

490 /* Process locked */

492

494 {

496 hdcmi->State = HAL_DCMI_STATE_SUSPENDED;
497
498 /* Disable Capture */
499 hdcmi->Instance->CR &= ~(DCMI_CR_CAPTURE);
500
501 /* Check if the DCMI capture effectively disabled */
502 do
503 {
504 if (count-- == 0U)
505 {
506 /* Update error code */
507 hdcmi->ErrorCode |= HAL_DCMI_ERROR_TIMEOUT;
508
509 /* Change DCMI state */
510 hdcmi->State = HAL_DCMI_STATE_READY;
511
512 status = HAL_TIMEOUT;
513 break;
514 }
515 }
516 while((hdcmi->Instance->CR & DCMI_CR_CAPTURE) != 0);
517 }
518 /* Process Unlocked */
519 __HAL_UNLOCK(hdcmi);
520
521 /* Return function status */
522 return status;

557void HAL_DCMI_IRQHandler(DCMI_HandleTypeDef *hdcmi)
558{
559 uint32_t isr_value = READ_REG(hdcmi->Instance->MISR);
560
561 /* Synchronization error interrupt management *******************************/
562 if((isr_value & DCMI_FLAG_ERRRI) == DCMI_FLAG_ERRRI)
563 {
564 /* Clear the Synchronization error flag */
565 __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_ERRRI);
566
567 /* Update error code */
568 hdcmi->ErrorCode |= HAL_DCMI_ERROR_SYNC;
569
570 /* Change DCMI state */
571 hdcmi->State = HAL_DCMI_STATE_ERROR;
572
573 /* Set the synchronization error callback */
574 hdcmi->DMA_Handle->XferAbortCallback = DCMI_DMAError;
575
576 /* Abort the DMA Transfer */
577 if (HAL_DMA_Abort_IT(hdcmi->DMA_Handle) != HAL_OK)
578 {
579 DCMI_DMAError(hdcmi->DMA_Handle);
580 }
581 }
582 /* Overflow interrupt management ********************************************/
583 if((isr_value & DCMI_FLAG_OVRRI) == DCMI_FLAG_OVRRI)
584 {
585 /* Clear the Overflow flag */
586 __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_OVRRI);
587
588 /* Update error code */
589 hdcmi->ErrorCode |= HAL_DCMI_ERROR_OVR;
590
591 /* Change DCMI state */
592 hdcmi->State = HAL_DCMI_STATE_ERROR;
593
594 /* Set the overflow callback */
595 hdcmi->DMA_Handle->XferAbortCallback = DCMI_DMAError;
596
597 /* Abort the DMA Transfer */
598 if (HAL_DMA_Abort_IT(hdcmi->DMA_Handle) != HAL_OK)
599 {
600 DCMI_DMAError(hdcmi->DMA_Handle);
601 }
602 }
603 /* Line Interrupt management ************************************************/
604 if((isr_value & DCMI_FLAG_LINERI) == DCMI_FLAG_LINERI)
605 {
606 /* Clear the Line interrupt flag */
607 __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_LINERI);
608
609 /* Line interrupt Callback */
610#if (USE_HAL_DCMI_REGISTER_CALLBACKS == 1)
611 /*Call registered DCMI line event callback*/
612 hdcmi->LineEventCallback(hdcmi);
613#else
614 HAL_DCMI_LineEventCallback(hdcmi);
615#endif /* USE_HAL_DCMI_REGISTER_CALLBACKS */
616 }
617 /* VSYNC interrupt management ***********************************************/
618 if((isr_value & DCMI_FLAG_VSYNCRI) == DCMI_FLAG_VSYNCRI)
619 {
620 /* Clear the VSYNC flag */
621 __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_VSYNCRI);
622
623 /* VSYNC Callback */
624#if (USE_HAL_DCMI_REGISTER_CALLBACKS == 1)
625 /*Call registered DCMI vsync event callback*/
626 hdcmi->VsyncEventCallback(hdcmi);
627#else
628 HAL_DCMI_VsyncEventCallback(hdcmi);
629#endif /* USE_HAL_DCMI_REGISTER_CALLBACKS */
630 }
631 /* FRAME interrupt management ***********************************************/
632 if((isr_value & DCMI_FLAG_FRAMERI) == DCMI_FLAG_FRAMERI)
633 {
634 /* When snapshot mode, disable Vsync, Error and Overrun interrupts */
635 if((hdcmi->Instance->CR & DCMI_CR_CM) == DCMI_MODE_SNAPSHOT)
636 {
637 /* Disable the Line, Vsync, Error and Overrun interrupts */
638 __HAL_DCMI_DISABLE_IT(hdcmi, DCMI_IT_LINE | DCMI_IT_VSYNC | DCMI_IT_ERR | DCMI_IT_OVR);
639 }
640
641 /* Disable the Frame interrupt */
642 __HAL_DCMI_DISABLE_IT(hdcmi, DCMI_IT_FRAME);
643
644 /* Frame Callback */
645#if (USE_HAL_DCMI_REGISTER_CALLBACKS == 1)
646 /*Call registered DCMI frame event callback*/
647 hdcmi->FrameEventCallback(hdcmi);
648#else
649 HAL_DCMI_FrameEventCallback(hdcmi);
650#endif /* USE_HAL_DCMI_REGISTER_CALLBACKS */
651 }
652}
