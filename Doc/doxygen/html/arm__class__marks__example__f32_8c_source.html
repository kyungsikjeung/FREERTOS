156/* ----------------------------------------------------------------------------------
157* Main f32 test function. It returns maximum marks secured and student number
158* ------------------------------------------------------------------------------- */
159
160int32_t main()
161{
162
163#ifndef USE_STATIC_INIT
164
165 arm_matrix_instance_f32 srcA;
166 arm_matrix_instance_f32 srcB;
167 arm_matrix_instance_f32 dstC;
168
169 /* Input and output matrices initializations */
170 arm_mat_init_f32(&srcA, numStudents, numSubjects, (float32_t *)testMarks_f32);
171 arm_mat_init_f32(&srcB, numSubjects, 1, (float32_t *)testUnity_f32);
172 arm_mat_init_f32(&dstC, numStudents, 1, testOutput);
173
174#else
175
176 /* Static Initializations of Input and output matrix sizes and array */
177 arm_matrix_instance_f32 srcA = {NUMSTUDENTS, NUMSUBJECTS, (float32_t *)testMarks_f32};
178 arm_matrix_instance_f32 srcB = {NUMSUBJECTS, 1, (float32_t *)testUnity_f32};
179 arm_matrix_instance_f32 dstC = {NUMSTUDENTS, 1, testOutput};
180
181#endif
182
183
184 /* ----------------------------------------------------------------------
185 *Call the Matrix multiplication process function
186 * ------------------------------------------------------------------- */
187 arm_mat_mult_f32(&srcA, &srcB, &dstC);
188
189 /* ----------------------------------------------------------------------
190 ** Call the Max function to calculate max marks among numStudents
191 ** ------------------------------------------------------------------- */
192 arm_max_f32(testOutput, numStudents, &max_marks, &student_num);
193
194 /* ----------------------------------------------------------------------
195 ** Call the Min function to calculate min marks among numStudents
196 ** ------------------------------------------------------------------- */
197 arm_min_f32(testOutput, numStudents, &min_marks, &student_num);
198
199 /* ----------------------------------------------------------------------
200 ** Call the Mean function to calculate mean
201 ** ------------------------------------------------------------------- */
202 arm_mean_f32(testOutput, numStudents, &mean);
203
204 /* ----------------------------------------------------------------------
205 ** Call the std function to calculate standard deviation
206 ** ------------------------------------------------------------------- */
207 arm_std_f32(testOutput, numStudents, &std);
208
209 /* ----------------------------------------------------------------------
210 ** Call the var function to calculate variance
211 ** ------------------------------------------------------------------- */
212 arm_var_f32(testOutput, numStudents, &var);
213
214#if defined(SEMIHOSTING)
215 printf("mean = %f, std = %f\n",mean,std);
216#endif
217
218#if !defined(SEMIHOSTING)
219 while (1); /* main function does not return */
220#endif
221}
