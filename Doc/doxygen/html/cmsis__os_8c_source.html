102/* Convert from CMSIS type osPriority to FreeRTOS priority number */
103static unsigned portBASE_TYPE makeFreeRtosPriority (osPriority priority)
104{
105 unsigned portBASE_TYPE fpriority = tskIDLE_PRIORITY;
106
107 if (priority != osPriorityError) {
108 fpriority += (priority - osPriorityIdle);
109 }
110
111 return fpriority;
112}
113
114#if (INCLUDE_uxTaskPriorityGet == 1)
115/* Convert from FreeRTOS priority number to CMSIS type osPriority */
116static osPriority makeCmsisPriority (unsigned portBASE_TYPE fpriority)
117{
118 osPriority priority = osPriorityError;
119
120 if ((fpriority - tskIDLE_PRIORITY) <= (osPriorityRealtime - osPriorityIdle)) {
121 priority = (osPriority)((int)osPriorityIdle + (int)(fpriority - tskIDLE_PRIORITY));
122 }
123
124 return priority;
125}

543osEvent osSignalWait (int32_t signals, uint32_t millisec)
544{
545 osEvent ret;
546
547#if( configUSE_TASK_NOTIFICATIONS == 1 )
548
549 TickType_t ticks;
550
551 ret.value.signals = 0;
552 ticks = 0;
553 if (millisec == osWaitForever) {
554 ticks = portMAX_DELAY;
555 }
556 else if (millisec != 0) {
557 ticks = millisec / portTICK_PERIOD_MS;
558 if (ticks == 0) {
559 ticks = 1;
560 }
561 }
562
563 if (inHandlerMode())
564 {
565 ret.status = osErrorISR; /*Not allowed in ISR*/
566 }
567 else
568 {
569 if(xTaskNotifyWait( 0,(uint32_t) signals, (uint32_t *)&ret.value.signals, ticks) != pdTRUE)
570 {
571 if(ticks == 0) ret.status = osOK;
572 else ret.status = osEventTimeout;
573 }
574 else if(ret.value.signals < 0)
575 {
576 ret.status = osErrorValue;
577 }
578 else ret.status = osEventSignal;
579 }
580#else
581 (void) signals;
582 (void) millisec;
583
584 ret.status = osErrorOS; /* Task Notification not supported */
585#endif
586
587 return ret;

887 int itemSize = 4 * ((pool_def->item_sz + 3) / 4);

905

907 for (i = 0; i < pool_def->pool_sz; i++) {
908 thePool->markers[i] = 0;
909 }
910 }

912 vPortFree(thePool->markers);
913 vPortFree(thePool);
914 thePool = NULL;
915 }
916 }
917 else {
918 vPortFree(thePool);

936void *osPoolAlloc (osPoolId pool_id)
937{
938 int dummy = 0;
939 void *p = NULL;
940 uint32_t i;
941 uint32_t index;
942
943 if (inHandlerMode()) {
944 dummy = portSET_INTERRUPT_MASK_FROM_ISR();
945 }
946 else {
947 vPortEnterCritical();
948 }
949
950 for (i = 0; i < pool_id->pool_sz; i++) {
951 index = (pool_id->currentIndex + i) % pool_id->pool_sz;
952
953 if (pool_id->markers[index] == 0) {
954 pool_id->markers[index] = 1;
955 p = (void *)((uint32_t)(pool_id->pool) + (index * pool_id->item_sz));
956 pool_id->currentIndex = index;
957 break;
958 }
959 }
960
961 if (inHandlerMode()) {
962 portCLEAR_INTERRUPT_MASK_FROM_ISR(dummy);
963 }
964 else {
965 vPortExitCritical();
966 }
967
968 return p;
969}

996osStatus osPoolFree (osPoolId pool_id, void *block)
997{
998 uint32_t index;
999
1000 if (pool_id == NULL) {
1001 return osErrorParameter;
1002 }
1003
1004 if (block == NULL) {
1005 return osErrorParameter;
1006 }
1007
1008 if (block < pool_id->pool) {
1009 return osErrorParameter;
1010 }
1011
1012 index = (uint32_t)block - (uint32_t)(pool_id->pool);
1013 if (index % pool_id->item_sz) {
1014 return osErrorParameter;
1015 }
1016 index = index / pool_id->item_sz;
1017 if (index >= pool_id->pool_sz) {
1018 return osErrorParameter;
1019 }
1020
1021 pool_id->markers[index] = 0;
1022
1023 return osOK;
1024}
